public static String fn_A(double[] a){
  String b="";
  for (int c=0; c < a.length; c++) {
    if (a[c] == (long)a[c]) {
      b+=(String.format("%+d%s ",(long)a[c],(c == (a.length - 1) ? "" : (c == (a.length - 2) ? " * x" : String.format(" * x^%d",(a.length - 1 - c))))));
    }
 else {
      b+=(String.format("%+f%s ",a[c],(c == (a.length - 1) ? "" : (c == (a.length - 2) ? " * x" : String.format(" * x^%d",(a.length - 1 - c))))));
    }
  }
  return b;
}

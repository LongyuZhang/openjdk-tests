private static boolean fn_A(Class<?> a,Class<?> b,boolean c){
  if (!c || a == b)   return true;
  if (a == Byte.class)   return false;
  if (a == Double.class)   return b == Float.class;
  if (a == Float.class)   return b == Double.class;
  if (a == Integer.class)   return b == Long.class || b == Short.class;
  if (a == Long.class)   return b == Integer.class || b == Short.class;
  if (a == Short.class)   return b == Long.class || b == Integer.class;
  return false;
}

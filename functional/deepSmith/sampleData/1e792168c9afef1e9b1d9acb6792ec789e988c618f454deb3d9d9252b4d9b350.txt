private static String fn_A(Object a){
  if (a instanceof boolean[]) {
    return "boolean";
  }
 else   if (a instanceof int[]) {
    return "int";
  }
 else   if (a instanceof long[]) {
    return "long";
  }
 else   if (a instanceof float[]) {
    return "float";
  }
 else   if (a instanceof double[]) {
    return "double";
  }
 else   if (a instanceof String[]) {
    return "string";
  }
 else {
    return "null";
  }
}

public static void fn_A(int[][] a){
  for (int b=0; b < a.length; b++) {
    if (b % 2 == 0) {
      for (int c=0; c < a[b].length; c++) {
        System.out.println(a[b][c]);
      }
    }
 else {
      for (int c=0; c < a[b].length; c++) {
        System.out.println(a[b][a[b].length - c - 1]);
      }
    }
  }
}

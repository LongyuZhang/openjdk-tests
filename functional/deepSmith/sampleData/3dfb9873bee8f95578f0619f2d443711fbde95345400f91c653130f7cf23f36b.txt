private static double[] fn_A(float[] a,int b,int c){
  double d=0;
  double e=0;
  double f=0;
  for (int g=0, h=0; g < c; g++) {
    double i=0;
    for (int j=0; j < b; j++) {
      final float k=a[h++];
      i+=k;
      d+=k * j;
    }
    f+=i;
    e+=i * g;
  }
  d=0.5 + d / f;
  e=0.5 + e / f;
  return new double[]{d,e};
}

public static byte[] fn_A(int a,int b){
  byte[] c=new byte[b];
  if (b == 1) {
    c[0]=(byte)(0xff & a);
  }
 else   if (b == 2) {
    c[0]=(byte)(0xff & a);
    c[1]=(byte)((0xff00 & a) >> 8);
  }
 else {
    c[0]=(byte)(0xff & a);
    c[1]=(byte)((0xff00 & a) >> 8);
    c[2]=(byte)((0xff0000 & a) >> 16);
    c[3]=(byte)((0xff000000 & a) >> 24);
  }
  return c;
}

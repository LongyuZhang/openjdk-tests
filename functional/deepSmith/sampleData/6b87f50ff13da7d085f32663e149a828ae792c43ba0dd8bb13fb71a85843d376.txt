public static int[] fn_A(int[] a,int[] b){
  int[] c=new int[a.length + b.length];
  int d=0, e=0, f=0;
  boolean g=false;
  boolean h=false;
  while (!g || !h) {
    if (e == a.length)     g=true;
    if (f == b.length)     h=true;
    if (g && h)     break;
    if (!g && h)     c[d++]=a[e++];
 else     if (!h && g)     c[d++]=b[f++];
 else     if (a[e] <= b[f])     c[d++]=a[e++];
 else     if (b[f] <= a[e])     c[d++]=b[f++];
  }
  return c;
}

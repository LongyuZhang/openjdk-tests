private static int fn_A(int[] a,int b){
  if (a.length == 0 || a == null)   return Integer.MAX_VALUE;
  int c=0;
  int d;
  for (int e=d=0; e < b; e++) {
    if (d == 0) {
      c=a[e];
      d=1;
    }
 else {
      if (c == a[e]) {
        d++;
      }
 else {
        d--;
      }
    }
  }
  return c;
}

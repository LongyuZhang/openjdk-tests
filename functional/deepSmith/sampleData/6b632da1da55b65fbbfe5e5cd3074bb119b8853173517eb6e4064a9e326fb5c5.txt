public static int[] fn_A(int[] a){
  if (a.length == 0)   return new int[]{-1,-1,Integer.MIN_VALUE};
  int b=0;
  int c=0;
  int d=a[0];
  int e=0;
  int f=d;
  for (int g=1; g < a.length; g++) {
    if (f < 0) {
      f=a[g];
      e=g;
    }
 else {
      f+=a[g];
    }
    if (f >= d) {
      d=f;
      b=e;
      c=g;
    }
  }
  return new int[]{b,c,d};
}

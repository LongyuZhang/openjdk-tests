private static String fn_A(final byte[] a){
  final int b=a[0] & 0xff;
switch (b) {
case 0:
    return "unloaded";
case 1:
  return "loaded";
case 2:
return "loading";
case 3:
return "error (during load process)";
case 4:
return "unloading";
case 5:
return "load completing";
default :
return "invalid load status " + b;
}
}

private static int[] fn_A(byte[] a,int b){
  int[] c=new int[a.length / 4 + 1];
  int d=c.length;
  int e=0;
  int f=b;
  for (int g=a.length % 4; g > 0; --g, e++)   f=(f << 8) | (a[e] & 0xff);
  c[--d]=f;
  while (d > 0)   c[--d]=a[e++] << 24 | (a[e++] & 0xff) << 16 | (a[e++] & 0xff) << 8 | (a[e++] & 0xff);
  return c;
}

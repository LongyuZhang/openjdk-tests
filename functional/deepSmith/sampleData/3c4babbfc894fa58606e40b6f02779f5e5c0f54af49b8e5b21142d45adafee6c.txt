static void fn_A(byte a[],int b[],int c,int d,int e,int f,int g,int h){
  if (g == 8) {
    int i=h * f + d * e;
    for (int j=0; j < d; ++j)     a[i + j]=(byte)b[j + c];
  }
 else   if (g == 16) {
    int i=h * f + d * e;
    for (int j=0; j < d; ++j)     a[i + j]=(byte)(b[j + c] >>> 8);
  }
 else {
    int i=h * f + e / (8 / g);
    int k=b[c] << (8 - g * (e % (8 / g)) - g);
    a[i]|=k;
  }
}

public static void fn_A(byte[] a,int b,int c,double d,int e,int f,byte[] g,int h,int i,int j,int k){
  for (int l=0; l < j; l++) {
    int m=b + l * c;
    int n=h + l * i;
    int o=m + k;
    for (; m < o; m++, n++) {
      int p=(int)Math.round((a[m] & 0xFF) * d);
      if (p < e)       p=e;
      if (p > f)       p=f;
      g[n]=(byte)p;
    }
  }
}

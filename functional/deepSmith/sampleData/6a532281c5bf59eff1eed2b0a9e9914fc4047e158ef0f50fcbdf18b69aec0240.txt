private static int fn_A(int a){
  int b=0;
  if ((a & 0xffff0000) == 0) {
    b=16;
    a<<=16;
  }
  if ((a & 0xff000000) == 0) {
    b+=8;
    a<<=8;
  }
  if ((a & 0xf0000000) == 0) {
    b+=4;
    a<<=4;
  }
  if ((a & 0xc0000000) == 0) {
    b+=2;
    a<<=2;
  }
  if ((a & 0x80000000) == 0) {
    b++;
    if ((a & 0x40000000) == 0)     return 32;
  }
  return b;
}

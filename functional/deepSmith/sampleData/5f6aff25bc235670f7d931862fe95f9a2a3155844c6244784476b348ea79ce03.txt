public static float[] fn_A(float[] a,float[] b){
  float[] c=new float[4];
  c[0]=a[3] * b[0] + a[0] * b[3] + a[1] * b[2] - a[2] * b[1];
  c[1]=a[3] * b[1] + a[1] * b[3] + a[2] * b[0] - a[0] * b[2];
  c[2]=a[3] * b[2] + a[2] * b[3] + a[0] * b[1] - a[1] * b[0];
  c[3]=a[3] * b[3] - a[0] * b[0] - a[1] * b[1] - a[2] * b[2];
  return c;
}

public static long fn_A(final double a,final int b,final long c,final int d){
  int e=d - b;
  double f=((double)c * a);
  if (e != 0) {
    if (e < 0) {
      for (int g=0; g < -e; g++) {
        f*=0.1;
      }
    }
 else {
      for (int g=0; g < e; g++) {
        f*=10.0;
      }
    }
  }
  return (long)f;
}

private static int fn_A(final char[] a,int b,int c,final char d){
  int e=b;
  int f=c - 1;
  while (e <= f) {
    final int g=e + ((f - e) >> 1);
    final char h=a[g];
    if (h < d) {
      e=g + 1;
    }
 else     if (h > d) {
      f=g - 1;
    }
 else {
      return g;
    }
  }
  return -e - 1;
}

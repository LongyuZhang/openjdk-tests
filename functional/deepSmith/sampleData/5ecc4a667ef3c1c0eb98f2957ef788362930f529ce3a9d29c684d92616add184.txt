private static void fn_A(int a,float[] b,float[][] c){
  if (a == 1) {
    c[0][1]=0.0f;
    c[0][2]=0.0f;
    c[0][3]=0.0f;
    return;
  }
  for (int d=0; d < a - 1; ++d) {
    c[d][1]=(c[d + 1][0] - c[d][0]) / (b[d + 1] - b[d]);
    c[d][2]=c[d][3]=0.0f;
  }
  c[a - 1][1]=c[a - 2][1];
  c[a - 1][2]=c[a - 1][3]=0.0f;
}

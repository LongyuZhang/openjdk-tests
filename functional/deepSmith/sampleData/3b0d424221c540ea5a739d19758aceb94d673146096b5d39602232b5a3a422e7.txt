static void fn_A(float a,float b,float c,float d,float e,float[] f,float[] g){
  int h=g.length;
  float i=0.0f;
  float j=0.0f;
  float k=0.0f;
  float l=0.0f;
  for (int m=0; m < h; ++m) {
    float n=f[m];
    float o=a * n + b * l + c * k - d * j - e * i;
    g[m]=o;
    i=j;
    j=o;
    k=l;
    l=n;
  }
}

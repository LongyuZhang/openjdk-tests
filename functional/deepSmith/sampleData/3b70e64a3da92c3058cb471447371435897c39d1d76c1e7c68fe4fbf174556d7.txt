public static boolean fn_A(int[] a){
  if (a == null || a.length < 2) {
    return false;
  }
  for (int b=1; b < a.length; b++) {
    int c=b - 1;
    int d=b;
    while (c >= 0) {
      if (a[d] < a[c]) {
        int e=a[c];
        a[c]=a[d];
        a[d]=e;
        d--;
        c--;
      }
 else {
        if (a[d] == a[c]) {
          return true;
        }
        break;
      }
    }
  }
  return false;
}

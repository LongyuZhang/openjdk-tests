public static byte[] fn_A(byte[] a){
  int b=0;
  byte[] c=null;
  for (int d=0; d < a.length; d++) {
    if (d + 3 <= a.length) {
      if (a[d] == 13 && a[d + 1] == 10 && a[d + 2] == 13 && a[d + 3] == 10) {
        b=d + 4;
      }
    }
  }
  if (b != 0 && b < a.length) {
    c=new byte[a.length - b];
    int e=0;
    for (int d=b; d < a.length; d++) {
      c[e]=a[d];
      e++;
    }
  }
  return c;
}

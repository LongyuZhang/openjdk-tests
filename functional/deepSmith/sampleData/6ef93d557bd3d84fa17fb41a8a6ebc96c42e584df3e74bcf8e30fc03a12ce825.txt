public static float[] fn_A(double a,long b,int c){
  float[] d=new float[c * 2];
  if (c % 2 == 0) {
    int e=(int)((Math.round((float)b / (float)a * (float)c)));
    for (int f=0; f < e; f++) {
      d[f]=1.0f;
      d[c - 1 - f]=1.0f;
    }
  }
 else {
    int e=(int)((Math.round((float)b / (float)a * (float)c + 0.5)));
    d[0]=1.0f;
    for (int f=1; f < e; f++) {
      d[f]=1.0f;
      d[c - f]=1.0f;
    }
  }
  return d;
}

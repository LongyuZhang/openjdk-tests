public static void fn_A(int[] a){
  int b=0;
  double c;
  for (int d=0; d < a.length; ++d)   b+=a[d];
  c=b * 1.0 / a.length;
  System.out.println("The average is " + c);
  for (int d=0; d < a.length; d++)   if (a[d] > c)   System.out.print(a[d] + "  ");
  System.out.println("are greater than the average");
}

public static int fn_A(int a){
  int b=0;
  if ((a & 0xFFFF0000) != 0) {
    a>>=16;
    b|=16;
  }
  if ((a & 0xFF00) != 0) {
    a>>=8;
    b|=8;
  }
  if ((a & 0xF0) != 0) {
    a>>=4;
    b|=4;
  }
  if ((a & 0xC) != 0) {
    a>>=2;
    b|=2;
  }
  if ((a & 0x2) != 0) {
    b|=1;
  }
  return b;
}

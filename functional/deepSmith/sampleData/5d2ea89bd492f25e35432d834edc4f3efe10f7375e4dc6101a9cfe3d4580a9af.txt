public static final void fn_A(final float[] a,final int b,final float[] c,final int d,final int e,final float[] f,final int g,final int[] h,final float[] i){
  int j, k, l, m=0, n=0;
  for (j=0; j < e; j++) {
    float o=.5f * f[j];
    for (k=0; k < d; k++)     o-=a[b + k] * c[m++];
    if (j < g || o < i[g - 1]) {
      for (l=g - 1; (l >= 1) && (l > n || o < i[l - 1]); l--) {
        i[l]=i[l - 1];
        h[l]=h[l - 1];
      }
      i[l]=o;
      h[l]=j;
      n++;
    }
  }
}

public static double[][] fn_A(double[][] a,double[][] b){
  if (a.length != b.length) {
    throw new IllegalArgumentException("m1 and m2 must have the same number of rows:" + a.length + " != "+ b.length);
  }
  double[][] c=new double[a.length][a[0].length + b[0].length];
  int d=0;
  for (int e=0; e < a.length; e++) {
    d=0;
    for (int f=0; f < a[e].length; f++) {
      c[e][d++]=a[e][f];
    }
    for (int f=0; f < b[e].length; f++) {
      c[e][d++]=b[e][f];
    }
  }
  return c;
}

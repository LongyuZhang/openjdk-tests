public static String fn_A(long a){
  long b=a / (24 * 60 * 60* 1000);
  long c=(a / (60 * 60 * 1000) - b * 24);
  long d=((a / (60 * 1000)) - b * 24 * 60 - c * 60);
  long e=(a / 1000 - b * 24 * 60* 60 - c * 60 * 60 - d * 60);
  long f=(a - b * 24 * 60* 60* 1000 - c * 60 * 60* 1000 - d * 60 * 1000 - e * 1000);
  return (b > 0 ? b + "," : "") + c + ":"+ d+ ":"+ e+ "."+ f;
}

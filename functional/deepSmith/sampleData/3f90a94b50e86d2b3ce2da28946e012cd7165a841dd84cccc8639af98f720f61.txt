public static char fn_A(int a){
  if (a < 0) {
    throw new UnsupportedOperationException("Exceeds range: " + a);
  }
 else   if (a < 26) {
    return (char)('A' + a);
  }
 else   if (a < 36) {
    return (char)('0' + (a - 26));
  }
 else {
    throw new UnsupportedOperationException("Exceeds range: " + a);
  }
}

private static String fn_A(final int a){
switch (a) {
case 31:
    return "Timer-control Code (TC), indicating that the programme identification information is to be ignored";
case 30:
  return "Recording Inhibit/Terminate code (RI/T), indicating that the transmission has no label and is for example, not intended to be recorded";
case 29:
return "Interruption code (INT), indicating a break in the programme, which will continue after a short interval";
case 28:
return "Continuation code, indicating possibly an erroneous transmission state. No action required";
default :
return "reserved for future use";
}
}

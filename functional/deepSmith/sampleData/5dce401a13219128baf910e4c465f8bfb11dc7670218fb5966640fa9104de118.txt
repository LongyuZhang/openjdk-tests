private static void fn_A(double[] a,int[] b,int c,int d){
  int e, f;
  int g;
  e=c;
  f=c * 2;
  while (f <= d) {
    if ((f < d) && (a[b[f - 1]] < a[b[f]])) {
      f+=1;
    }
    if (a[b[e - 1]] < a[b[f - 1]]) {
      g=b[e - 1];
      b[e - 1]=b[f - 1];
      b[f - 1]=g;
    }
    e=f;
    f*=2;
  }
}

public static boolean fn_A(boolean[] a,boolean[] b){
  boolean c;
  c=a[0] != b[0];
  if (a.length != b.length)   throw new IllegalArgumentException("Splits must be of the same length!");
  for (int d=0; d < a.length; d++) {
    if (c) {
      if (a[d] == b[d])       return false;
    }
 else {
      if (a[d] != b[d])       return false;
    }
  }
  return true;
}

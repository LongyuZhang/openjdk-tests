public static int fn_A(int a,int[] b,int c,int d){
  if (b == null) {
    return 1;
  }
  if (d == -1) {
    return 1;
  }
  int e=c, f;
  while (c <= d) {
    e=c + (d - c) / 2;
    if (a < (f=b[e])) {
      d=e - 1;
    }
 else     if (a > f) {
      c=e + 1;
    }
 else {
      return e + 1;
    }
  }
  if (a < b[e]) {
    return e + 1;
  }
  return e + 2;
}

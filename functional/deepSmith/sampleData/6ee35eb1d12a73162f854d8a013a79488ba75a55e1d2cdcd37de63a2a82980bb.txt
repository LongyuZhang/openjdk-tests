private static void fn_A(int[] a){
  int b=a[0];
  int c=1;
  int d=a.length;
  int[] e=new int[10];
  for (int f=1; f < d; f++) {
    if (a[f] > b)     b=a[f];
  }
  while (b / c > 0) {
    int[] g=new int[10];
    for (int f=0; f < d; f++)     g[(a[f] / c) % 10]++;
    for (int f=1; f < 10; f++)     g[f]+=g[f - 1];
    for (int f=d - 1; f >= 0; f--)     e[--g[(a[f] / c) % 10]]=a[f];
    for (int f=0; f < d; f++)     a[f]=e[f];
    c*=10;
  }
}

private static int fn_A(int a){
  if (a <= 5) {
    return 0;
  }
 else   if (a >= 7 && a <= 11) {
    return 1;
  }
 else   if (a >= 13 && a <= 17) {
    return 2;
  }
 else   if (a >= 19 && a <= 23) {
    return 3;
  }
 else   if (a >= 25 && a <= 29) {
    return 4;
  }
 else   if (a >= 31 && a <= 35) {
    return 5;
  }
 else   if (a >= 37 && a <= 38) {
    return 6;
  }
  return 0;
}

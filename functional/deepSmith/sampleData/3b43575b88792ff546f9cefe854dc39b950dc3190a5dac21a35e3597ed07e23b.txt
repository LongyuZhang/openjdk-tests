public static int fn_A(int[] a){
  int b=0;
  int c=0;
  for (int d=0; d < a.length; d++) {
    if (b < a[d]) {
      b=a[d];
    }
    if (c > a[d]) {
      c=a[d];
    }
  }
  int e=b - c + 1;
  int[] f=new int[e];
  for (int d=0; d < a.length; d++) {
    int g=a[d];
    f[g - c]+=1;
  }
  for (int d=0; d < f.length; d++) {
    if (f[d] == 1) {
      return d + c;
    }
  }
  return -1;
}

public static void fn_A(long[][] a,double b,int c){
  int d=a.length - 1;
  for (int e=0; e < c - 1; e++) {
    long f=0;
    for (int g=0; g < d; g++) {
      f+=a[g][e];
    }
    long h=0, i=0, j=0;
    if (d == 0)     h=f;
 else     h=f / d;
    for (int g=0; g < d; g++) {
      if (a[g][e] < b * h) {
        i+=a[g][e];
        j++;
      }
    }
    if (j != 0)     a[d][e]=i / j;
  }
}

protected static void fn_A(char[] a,int[] b){
  int c=a.length - 1;
  int d=0;
  int e=0;
  while (d <= c) {
    if (a[d] != '/')     break;
    a[d]='\0';
    d++;
  }
  while (d <= c) {
    b[e++]=d++;
    while (d <= c) {
      if (a[d++] != '/')       continue;
      a[d - 1]='\0';
      while (d <= c) {
        if (a[d] != '/')         break;
        a[d++]='\0';
      }
      break;
    }
  }
  if (e != b.length)   throw new InternalError();
}

public static int fn_A(int[] a){
  if (a == null || a.length == 0) {
    return -1;
  }
  if (a.length == 1 || a[0] < a[1]) {
    return 0;
  }
  if (a[a.length - 1] < a[a.length - 2]) {
    return a.length - 1;
  }
  int b=0;
  int c=1;
  int d=a.length - 2;
  while (c < d) {
    b=(c + d) / 2;
    if (a[b - 1] < a[b]) {
      d=b - 1;
    }
 else     if (a[b + 1] < a[b]) {
      c=b + 1;
    }
 else {
      return b;
    }
  }
  return c;
}

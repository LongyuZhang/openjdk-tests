public static void fn_A(byte[] a){
  int b=a.length / 4;
  int c=(b - 1) * 4;
  int d;
  d=(a[--c] & 0xFF) << 24;
  d|=(a[--c] & 0xFF) << 16;
  d|=(a[--c] & 0xFF) << 8;
  d|=(a[--c] & 0xFF);
  int e;
  while (c > 4) {
    a[--c]^=(d >> 24);
    e=(a[c] & 0xFF) << 24;
    a[--c]^=(d >> 16);
    e+=(a[c] & 0xFF) << 16;
    a[--c]^=(d >> 8);
    e+=(a[c] & 0xFF) << 8;
    a[--c]^=d;
    e+=(a[c] & 0xFF);
    d=d - e;
  }
}

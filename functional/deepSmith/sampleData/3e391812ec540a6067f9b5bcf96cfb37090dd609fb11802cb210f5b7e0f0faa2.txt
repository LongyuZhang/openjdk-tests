private static int fn_A(byte a){
  if (a < 0)   return 0;
  if (a < 1)   return 8;
 else   if (a < 2)   return 7;
 else   if (a < 4)   return 6;
 else   if (a < 8)   return 5;
 else   if (a < 16)   return 4;
 else   if (a < 32)   return 3;
 else   if (a < 64)   return 2;
 else   if (a < 128)   return 1;
 else   return 0;
}

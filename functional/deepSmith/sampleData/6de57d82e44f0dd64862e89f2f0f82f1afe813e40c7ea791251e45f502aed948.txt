private static void fn_A(byte[] a,int b,byte[] c,int d,int e){
  if (b >= 0 && d >= 0 && e >= 0 && e <= a.length - b && e <= c.length - d) {
    if (a != c || b > d || b + e <= d) {
      for (int f=0; f < e; ++f) {
        c[d + f]=a[b + f];
      }
    }
 else {
      for (int f=e - 1; f >= 0; --f) {
        c[d + f]=a[b + f];
      }
    }
  }
 else {
    throw new ArrayIndexOutOfBoundsException();
  }
}

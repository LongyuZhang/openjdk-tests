public static int fn_A(int a,int b){
  boolean c=false;
  if (a < 0) {
    c=true;
    a=-a;
  }
  if (b < 0) {
    b=-b;
  }
  int d=0;
  int e=0;
  for (int f=0; f < 32; ++f) {
    d<<=1;
    e<<=1;
    if ((a & 0x80000000) != 0) {
      e|=1;
    }
    a<<=1;
    if (e >= b) {
      e-=b;
      d|=1;
    }
  }
  if (c) {
    e=-e;
  }
  return e;
}

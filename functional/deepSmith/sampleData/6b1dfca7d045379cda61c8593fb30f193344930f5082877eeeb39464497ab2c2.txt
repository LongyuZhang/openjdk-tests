private static char fn_A(byte[] a,int b,int c){
  char d=0;
  if (c > 0) {
    while (true) {
      char e=(char)(a[b++]);
      if (e <= '9') {
        if (e < '0')         break;
        d+=(e & 0xF);
      }
      if (--c == 0)       return d;
      if ((d >= 6553) && ((d > 6553) || (a[b + 1] > '5')))       break;
      d=(char)((d << 1) + (d << 3));
    }
  }
  return '\uffff';
}

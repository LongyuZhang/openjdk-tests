protected static Boolean fn_A(Object a){
  if (a instanceof Boolean) {
    return (Boolean)a;
  }
  if (a instanceof String) {
    if ("true".equalsIgnoreCase((String)a)) {
      return Boolean.TRUE;
    }
    if ("false".equalsIgnoreCase((String)a)) {
      return Boolean.FALSE;
    }
  }
  return null;
}

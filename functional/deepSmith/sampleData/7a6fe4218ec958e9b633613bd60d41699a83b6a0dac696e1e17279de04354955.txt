public static double[] fn_A(double a[][],double b[]){
  int c=a.length;
  double d[]=new double[b.length];
  for (int e=0; e < c; e++) {
    d[e]=b[e];
    for (int f=0; f < e; f++) {
      d[e]-=a[e][f] * d[f];
    }
  }
  double g[]=new double[d.length];
  for (int h=0; h < c; h++) {
    g[h]=d[h] / a[h][h];
  }
  double i[]=new double[g.length];
  for (int e=c - 1; e >= 0; e--) {
    i[e]=g[e];
    for (int f=e + 1; f < c; f++) {
      i[e]-=a[f][e] * i[f];
    }
  }
  return i;
}

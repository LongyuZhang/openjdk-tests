public static double[][] fn_A(double[][] a,double[][] b){
  final int c=a[0].length;
  final int d=a.length;
  final int e=b[0].length;
  final double[][] f=new double[d][e];
  for (int g=0; g < d; g++) {
    final double[] h=f[g];
    final double[] i=a[g];
    for (int j=0; j < e; j++) {
      for (int k=0; k < c; k++) {
        h[j]+=i[k] * b[k][j];
      }
    }
  }
  return f;
}

public static String fn_A(long a){
  String b=null;
  int c=(int)(a / 1000);
  int d=c / 60;
  int e=d / 60;
  if (e >= 0 && e < 10)   b="0" + e + ":";
 else   b=e + ":";
  if (e > 0)   d=d % 60;
  if (d >= 0 && d < 10)   b+="0" + d + ":";
 else   b+=d + ":";
  c=c % 60;
  if (c >= 0 && c < 10)   b+="0" + c;
 else   b+=c;
  return b;
}

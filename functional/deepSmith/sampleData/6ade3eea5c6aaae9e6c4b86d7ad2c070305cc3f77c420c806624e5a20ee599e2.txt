public static float[] fn_A(float a,float b,float c,float d,float e,float f){
  float[] g=new float[16];
  float h=(b - a), i=(d - c), j=(e - f);
  g[0]=(2 * e) / h;
  g[4]=0;
  g[8]=(b + a) / h;
  g[12]=0;
  g[1]=0;
  g[5]=(2 * e) / i;
  g[9]=(d + c) / i;
  g[13]=0;
  g[2]=0;
  g[6]=0;
  g[10]=(e + f) / j;
  g[14]=(2 * e * f) / j;
  g[3]=0;
  g[7]=0;
  g[11]=-1;
  g[15]=0;
  return g;
}

public static int fn_A(byte[] a,byte[] b,int c,int d,int e){
  for (int f=0; f < e; ++f) {
    if (f + c == a.length)     return f + d == b.length ? 0 : -1;
 else     if (f + d == b.length)     return 1;
 else     if ((0xff & a[f + c]) > (0xff & b[f + d]))     return 1;
 else     if ((0xff & a[f + c]) < (0xff & b[f + d]))     return -1;
  }
  return 0;
}

private static double[] fn_A(int a,boolean b){
  double[] c=new double[a + 1];
  c[0]=1;
  c[1]=a;
  for (int d=2; d < a / 2 + 1; ++d) {
    c[d]=(a - d + 1) * c[d - 1] / d;
    c[a - d]=c[d];
  }
  c[a - 1]=a;
  c[a]=1;
  if (!b) {
    for (int d=1; d < a + 1; d+=2)     c[d]=-c[d];
  }
  return c;
}

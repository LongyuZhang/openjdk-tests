private static void fn_A(final long a,final long b,final long c){
  if (b < 0L) {
    throw new ArrayIndexOutOfBoundsException("Start index (" + b + ") is negative");
  }
 else   if (b > c) {
    throw new IllegalArgumentException("Start index (" + b + ") is greater than end index ("+ c+ ")");
  }
 else   if (c > a) {
    throw new ArrayIndexOutOfBoundsException("End index (" + c + ") is greater than bit vector length ("+ a+ ")");
  }
}

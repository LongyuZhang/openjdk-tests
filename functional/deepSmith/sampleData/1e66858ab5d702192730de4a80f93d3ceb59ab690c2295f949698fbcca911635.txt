public static int[][] fn_A(int[][] a){
  int b=a.length;
  int c=a[0].length;
  int[][] d=new int[b][c];
  int e=0;
  int f=0;
  int g=0;
  for (int h=0; h < b; h++) {
    for (int i=0; i < c; i++) {
      if (a[h][i] > 0) {
        e+=h;
        f+=i;
        g++;
      }
    }
  }
  double j=0;
  double k=0;
  if (g != 0) {
    j=e * 1.0 / g;
    k=f * 1.0 / g;
    d[(int)j][(int)k]=255;
  }
  return d;
}

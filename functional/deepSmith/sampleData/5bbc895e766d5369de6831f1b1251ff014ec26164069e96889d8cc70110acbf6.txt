private static int fn_A(final char[] a,final int b,final int c,final char[] d){
  int e=b;
  int f, g;
  char h;
  if (a.length == 0) {
    return c;
  }
  h=d[0];
  while (e < c) {
    if (h == a[e]) {
      g=e;
      f=0;
      while (e < c && f < d.length) {
        if (d[f] != a[e]) {
          break;
        }
        ++f;
        ++e;
      }
      e=g;
      if (f >= d.length) {
        break;
      }
    }
    ++e;
  }
  return e;
}

public static boolean fn_A(int a){
  if (a <= 0) {
    return false;
  }
  if (a == 1) {
    return true;
  }
  int b=a;
  while (b > 1) {
    int c=0;
    if ((c=b % 2) == 0) {
      b=b / 2;
      continue;
    }
    if ((c=b % 3) == 0) {
      b=b / 3;
      continue;
    }
    if ((c=b % 5) == 0) {
      b=b / 5;
      continue;
    }
    return false;
  }
  return true;
}

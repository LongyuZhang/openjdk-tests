public static void fn_A(final double[] a,final double[] b,final double[] c,int d,int e,int f,final int g,final int h,final int i){
  int j=d;
  int k=f;
  for (int l=0; l < g; l++, k+=i) {
    int m=e;
    final int n=k + i;
    final int o=j + h;
    while (j != o) {
      double p=a[j++];
      int q=k;
      while (q != n) {
        c[q++]-=p * b[m++];
      }
    }
  }
}

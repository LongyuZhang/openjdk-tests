private static float fn_A(float[][][] a,float b,float c,float d){
  float e=0.0f;
  for (int f=3; f >= 0; --f) {
    float g=0.0f;
    for (int h=3; h >= 0; --h) {
      float i=0.0f;
      for (int j=3; j >= 0; --j) {
        i=a[f][h][j] + b * i;
      }
      g=i + c * g;
    }
    e=g + d * e;
  }
  return e;
}

public static int fn_A(int[] a,int b){
  int[] c=new int[a.length];
  c[a[0]]=1;
  int d, e;
  for (d=1; d < a.length; ++d) {
    int f=a[d];
    for (e=1; e <= b; ++e)     if (c[f - e] != 0)     break;
    if (e > b && c[f - e] != 0)     return d + 1;
    for (e=1; e <= b; ++e)     if (c[f + e] != 0)     break;
    if (e > b && c[f + e] != 0)     return d + 1;
    c[a[d]]=1;
  }
  return -1;
}

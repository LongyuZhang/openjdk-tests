private static boolean fn_A(int a,int b,int c,int[][] d){
  for (int e=0; e < 9; ++e) {
    if (d[a][e] == c || d[e][b] == c)     return false;
  }
  int f=a / 3 * 3, g=b / 3 * 3;
  for (int h=0; h < 3; ++h) {
    for (int e=0; e < 3; ++e) {
      if (d[f + h][g + e] == c)       return false;
    }
  }
  return true;
}

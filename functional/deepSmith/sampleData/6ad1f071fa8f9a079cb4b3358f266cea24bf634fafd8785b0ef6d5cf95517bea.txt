private static void fn_A(float[] a,int[] b,int c,int d,int e,int f,int[] g,int h,boolean i){
  int j=c;
  int k=e;
  int l=h;
  while (j <= d && k <= f) {
    if (i) {
      if (a[b[j]] <= a[b[k]])       g[l++]=b[j++];
 else       g[l++]=b[k++];
    }
 else {
      if (a[b[j]] >= a[b[k]])       g[l++]=b[j++];
 else       g[l++]=b[k++];
    }
  }
  while (j <= d)   g[l++]=b[j++];
  while (k <= f)   g[l++]=b[k++];
}

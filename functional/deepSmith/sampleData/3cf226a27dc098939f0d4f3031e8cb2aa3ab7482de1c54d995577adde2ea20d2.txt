public static byte[] fn_A(byte[] a,int b) throws IllegalArgumentException {
  if (b < 0) {
    throw new IllegalArgumentException("Value for variable \"places\" cannot be negative.");
  }
  byte[] c=new byte[a.length];
  for (int d=a.length - b - 1; d >= 0; d--) {
    c[d + b]=a[d];
  }
  return c;
}

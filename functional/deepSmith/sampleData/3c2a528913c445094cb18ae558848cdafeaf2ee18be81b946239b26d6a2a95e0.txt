protected static Class fn_A(Class a){
  if (a == Integer.TYPE) {
    a=Integer.class;
  }
 else   if (a == Byte.TYPE) {
    a=Byte.class;
  }
 else   if (a == Long.TYPE) {
    a=Long.class;
  }
 else   if (a == Double.TYPE) {
    a=Double.class;
  }
 else   if (a == Float.TYPE) {
    a=Float.class;
  }
 else   if (a == Boolean.TYPE) {
    a=Boolean.class;
  }
 else   if (a == Character.TYPE) {
    a=Character.class;
  }
 else   if (a == Short.TYPE) {
    a=Short.class;
  }
  return a;
}

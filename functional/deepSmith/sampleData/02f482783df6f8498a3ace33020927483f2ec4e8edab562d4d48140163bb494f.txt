private static void fn_A(int a[],int b,int c){
  int d=(b + c) / 2;
  int e[]=new int[d - b + 1];
  int f[]=new int[c - d];
  int g, h=0, i=0, j=0;
  for (g=b; g <= d; g++, h++)   e[h]=a[g];
  h=0;
  for (; g <= c; g++, h++)   f[h]=a[g];
  for (g=b; g <= c && i < d - b + 1 && j < c - d; g++) {
    if (e[i] < f[j]) {
      a[g]=e[i];
      i++;
    }
 else {
      a[g]=f[j];
      j++;
    }
  }
  for (; i < d - b + 1; i++)   a[g++]=e[i];
  for (; j < c - d; j++)   a[g++]=f[j];
}

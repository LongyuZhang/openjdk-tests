public static int[] fn_A(int[] a,int b){
  int[] c={-1,-1};
  if (a.length == 0)   return c;
  int d=0;
  int e=a.length - 1;
  while (d < e) {
    int f=(d + e) / 2;
    if (a[f] < b) {
      d=f + 1;
    }
 else {
      e=f;
    }
  }
  if (a[d] != b)   return c;
  c[0]=d;
  e=a.length - 1;
  while (d < e) {
    int f=(d + e) / 2 + 1;
    if (a[f] > b) {
      e=f - 1;
    }
 else {
      d=f;
    }
  }
  if (a[d] != b)   return c;
  c[1]=e;
  return c;
}

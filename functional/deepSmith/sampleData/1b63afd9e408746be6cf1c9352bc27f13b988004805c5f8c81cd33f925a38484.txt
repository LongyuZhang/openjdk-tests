public static int fn_A(char a){
  if (a >= '0' && a <= '9') {
    return a - 48;
  }
  if (a >= 'A' && a <= 'F') {
    return (a - 65) + 10;
  }
  if (a >= 'a' && a <= 'f') {
    return (a - 97) + 10;
  }
  throw new IllegalArgumentException("Input string may only contain hex digits, but found '" + a + "'");
}

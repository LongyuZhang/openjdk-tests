public static String fn_A(long a){
  if (a < 0)   return "0";
  long b=0, c=0, d=0;
  if (a > 60) {
    c=a / 60;
    if (c > 24) {
      b=c / 24;
      if (b > 0)       c=c % 24;
    }
    d=a % 60;
  }
 else   d=a;
  if (b > 0)   return b + "d " + c+ "h "+ d+ "m";
 else   if (c > 0)   return c + "h " + d+ "m";
 else   if (d >= 0)   return d + "m";
 else   return "0";
}

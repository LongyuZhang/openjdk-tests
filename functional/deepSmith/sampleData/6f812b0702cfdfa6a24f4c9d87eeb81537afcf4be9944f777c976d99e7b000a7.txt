public static double[] fn_A(double[] a){
  double b=a[0];
  double c=a[1];
  double d=a[2];
  if (a[0] < 0.5)   b=0.5;
 else   if (a[0] > 1000)   b=1000;
  if (a[1] < -Math.PI / 4)   c=-Math.PI / 4;
 else   if (a[1] > Math.PI / 4)   c=Math.PI / 4;
  if (a[2] < -Math.PI / 12)   d=-Math.PI / 12;
 else   if (a[2] > Math.PI / 12)   d=Math.PI / 12;
  if (a[0] < 30) {
    c=0.0;
    d=0.0;
  }
  a[0]=b;
  a[1]=c;
  a[2]=d;
  return a;
}

public static byte[] fn_A(int a){
  byte[] b=new byte[9];
  if (a < 0) {
    throw new UnsupportedOperationException("Negative values cannot be encoded as varint.");
  }
  int c=0;
  while (true) {
    int d=(a & 0x7f);
    a>>=7;
    if (a != 0) {
      b[c]=(byte)(d | 0x80);
      c+=1;
    }
 else {
      b[c]=(byte)d;
      c+=1;
      break;
    }
  }
  byte[] e=new byte[c];
  for (int f=0; f < c; f++) {
    e[f]=b[f];
  }
  return e;
}

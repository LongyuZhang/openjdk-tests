  double h, i, j;
  h=c + (d - g + (e + f / 60.0) / 60.0) / 24.0;
  if (b < 3) {
    b+=12;
    a--;
  }
  i=Math.floor(365.25 * (a + 4716.0)) + Math.floor(30.6001 * (b + 1)) + h - 1524.5;
  if (i > 2299160.0) {

    i+=(2 - j + Math.floor(j / 4));
  }

}

public static final double[][] fn_A(double a[][],double b[][],double c[][]){
  int d=a.length, e=b.length;
  int f=a[0].length;
  assert(a[0].length == b[0].length);
  if (c == null) {
    c=new double[d][e];
  }
  assert(c.length == d && c[0].length == e);
  for (int g=0; g < d; g++) {
    for (int h=0; h < e; h++) {
      double i=0;
      for (int j=0; j < f; j++) {
        i+=a[g][j] * b[h][j];
      }
      c[g][h]=i;
    }
  }
  return c;
}

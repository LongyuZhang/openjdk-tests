protected static boolean fn_A(int a){
  if ((a < 33) || (a > 126)) {
    return false;
  }
switch (a) {
case '"':
case '(':
case ')':
case ',':
case '/':
case ':':
case ';':
case '<':
case '=':
case '>':
case '?':
case '@':
case '[':
case '\\':
case ']':
case '{':
case '}':
    return false;
}
return true;
}

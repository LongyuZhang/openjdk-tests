public static double[] fn_A(double[][] a){
  int b=a.length;
  int c=a[0].length;
  double[] d=new double[c * c];
  double[] e=new double[c];
  for (int f=0; f < b; f++) {
    for (int g=0; g < c; g++) {
      e[g]+=a[f][g];
    }
  }
  for (int g=0; g < c; g++) {
    e[g]/=b;
  }
  for (int f=0; f < b; f++) {
    for (int h=0; h < c; h++) {
      for (int i=0; i < c; i++) {
        d[(c * h) + i]+=(a[f][h] - e[h]) * (a[f][i] - e[i]);
      }
    }
  }
  return d;
}

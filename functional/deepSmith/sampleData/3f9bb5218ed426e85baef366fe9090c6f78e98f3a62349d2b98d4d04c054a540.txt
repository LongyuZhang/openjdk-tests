public static int fn_A(int[] a,int b){
  if (a == null || a.length == 0 || b < 0) {
    return 0;
  }
  int[][] c=new int[a.length][b + 1];
  for (int d=0; d < a.length; d++) {
    c[d][0]=1;
  }
  for (int e=1; a[0] * e <= b; e++) {
    c[0][a[0] * e]=1;
  }
  int f=0;
  for (int d=1; d < a.length; d++) {
    for (int e=1; e <= b; e++) {
      f=0;
      for (int g=0; e - a[d] * g >= 0; g++) {
        f+=c[d - 1][e - a[d] * g];
      }
      c[d][e]=f;
    }
  }
  return c[a.length - 1][b];
}

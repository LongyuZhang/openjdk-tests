public static short fn_A(int a){
  short b=8;
  if (a > 79 && a < 90) {
    b=1;
  }
 else   if (a > 69 && a < 80) {
    b=2;
  }
 else   if ((a > 39 && a < 50) || (a > 59 && a < 70)) {
    b=3;
  }
 else   if ((a > 30 && a < 36) || (a > 49 && a < 56)) {
    b=4;
  }
 else   if (a == 30) {
    b=5;
  }
 else   if (a == 36 || a == 37) {
    b=6;
  }
 else   if ((a > 0 && a < 30) || (a > 89 && a < 100)) {
    b=7;
  }
 else   if (a == 0) {
    b=8;
  }
  return (short)b;
}

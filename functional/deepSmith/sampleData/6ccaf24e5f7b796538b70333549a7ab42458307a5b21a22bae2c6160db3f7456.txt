public static void main(String[] a){
  int b=4;
  int c=++b;
  b=4;
  int d=b++;
  System.out.println("v is " + b);
  System.out.println("++v is " + c);
  System.out.println("v++ is " + d);
  int e=17, f=17, g=18;
  boolean h=(++e == g);
  boolean i=(f++ == g);
  System.out.println("First compare is " + h);
  System.out.println("Second compare is " + i);
}

public static boolean fn_A(byte[] a,byte[] b){
  if (a == null || b == null) {
    if (a == b)     return true;
 else     return false;
  }
 else   if (b.length != a.length) {
    return false;
  }
  for (int c=0, d=a.length; c < d; c++) {
    if (a[c] != b[c])     return false;
  }
  return true;
}

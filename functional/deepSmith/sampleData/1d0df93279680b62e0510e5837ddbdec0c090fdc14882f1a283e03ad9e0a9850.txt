public static void fn_A(Comparable[][] a){
  int b=a.length;
  if (b == 0) {
    throw new IllegalArgumentException("martix error");
  }
  int c=a[0].length;
  for (int d=0; d < b; d++) {
    for (int e=0; e < c; e++) {
      if (e == c - 1) {
        System.out.print(a[d][e]);
      }
 else {
        System.out.print(a[d][e] + ",");
      }
    }
    System.out.println();
  }
}

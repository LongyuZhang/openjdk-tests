private static void fn_A(int a,float[][] b,int c,int d,int e){
  int f=2 * a;
  for (int g=0; g < c; ++g) {
    float[] h=b[d];
    float[] i=b[e];
    for (int j=0; j < f; j+=2) {
      float k=h[j] - i[j];
      float l=h[j + 1] - i[j + 1];
      h[j]=h[j] + i[j];
      h[j + 1]=h[j + 1] + i[j + 1];
      i[j]=k;
      i[j + 1]=l;
    }
    int m=e + 1;
    e=d + 1;
    d=m;
  }
}

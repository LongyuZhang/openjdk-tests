public static void fn_A(int[][] a,int[][] b){
  int[][] c=new int[a.length][a[0].length];
  c[0][0]=0;
  for (int d=1; d < a.length; ++d) {
    c[0][d]=c[0][d - 1] + a[0][d];
  }
  for (int d=1; d < b.length; ++d) {
    c[d][0]=c[d - 1][0] + b[d][0];
  }
  for (int d=1; d < a.length; ++d) {
    for (int e=1; e < a[d].length; ++e) {
      c[d][e]=Math.min(c[d][e - 1] + a[d][e],c[d - 1][e] + b[d][e]);
    }
  }
  System.out.println(c[a.length - 1][a.length - 1]);
}

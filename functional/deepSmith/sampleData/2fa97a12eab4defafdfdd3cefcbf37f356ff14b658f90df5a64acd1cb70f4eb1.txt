public static int fn_A(long a){
  if (a < 0) {
    return 0;
  }
  if (a == 0) {
    return 64;
  }
  int b=1;
  int c=(int)(a >>> 32);
  if (c == 0) {
    b+=32;
    c=(int)a;
  }
  if (c >>> 16 == 0) {
    b+=16;
    c<<=16;
  }
  if (c >>> 24 == 0) {
    b+=8;
    c<<=8;
  }
  if (c >>> 28 == 0) {
    b+=4;
    c<<=4;
  }
  if (c >>> 30 == 0) {
    b+=2;
    c<<=2;
  }
  return b - (c >>> 31);
}

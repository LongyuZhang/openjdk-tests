public static int fn_A(int[] a){
  int b=a[0];
  int c=-1;
  boolean d=true;
  for (int e=1; e < a.length; e++) {
    if (!(b == a[e])) {
      if (d) {
        c=e;
        d=false;
      }
 else {
        d=true;
        break;
      }
    }
  }
  if (d) {
    b=a[1];
    for (int e=0; e < a.length; e++) {
      if (!(b == a[e])) {
        if (d) {
          c=e;
          d=false;
        }
 else {
          return -1;
        }
      }
    }
  }
  return c;
}

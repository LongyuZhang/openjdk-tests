public static double fn_A(double[] a){
  double b=0, c=0;
  if (a.length <= 1) {
    return 0;
  }
  for (int d=0; d < a.length; d++) {
    b+=a[d];
    c+=(a[d] * a[d]);
  }
  double e=(c - (b * b / (double)a.length)) / (double)(a.length - 1);
  if (e < 0) {
    return 0;
  }
 else {
    return e;
  }
}

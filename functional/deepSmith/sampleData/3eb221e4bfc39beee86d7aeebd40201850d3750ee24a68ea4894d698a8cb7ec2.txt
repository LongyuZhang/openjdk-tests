private static int fn_A(byte[] a){
  final int b=a.length;
  int c=0;
  while (c < b) {
    byte d=a[c++];
    if (d == (byte)' ' || d == (byte)'\n' || d == (byte)'\r' || d == (byte)'\t') {
      int e=c - 1;
      while (c < b) {
        d=a[c++];
        if (d != (byte)' ' && d != (byte)'\n' && d != (byte)'\r' && d != (byte)'\t') {
          a[e++]=d;
        }
      }
      return e;
    }
  }
  return b;
}

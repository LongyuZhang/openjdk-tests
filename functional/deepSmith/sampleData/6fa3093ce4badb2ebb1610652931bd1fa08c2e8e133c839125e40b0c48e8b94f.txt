public static byte fn_A(byte[][] a,int[] b){
  int c=0;
  int d=0;
  byte e=a[0][0];
  int f=a.length;
  for (int g=0; g < f; ++g) {
    byte[] h=a[g];
    int i=h.length;
    for (int j=0; j < i; ++j) {
      if (h[j] < e) {
        e=h[j];
        d=g;
        c=j;
      }
    }
  }
  if (b != null) {
    b[0]=c;
    b[1]=d;
  }
  return e;
}

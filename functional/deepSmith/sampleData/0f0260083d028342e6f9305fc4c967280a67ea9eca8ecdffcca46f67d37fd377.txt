public static void main(String[] a){
  if (a.length < 3) {
    System.out.println("Usage: java RuleOfThree <a> <b> <c>");
    System.exit(-1);
  }
  int b=Integer.parseInt(a[0]);
  int c=Integer.parseInt(a[1]);
  int d=Integer.parseInt(a[2]);
  int e=(c * d) / b;
  System.out.println("x = " + e);
}

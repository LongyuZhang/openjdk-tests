  int c, d;
  if (b == 0) {
    return a;
  }
  if (a < b) {
    c=a;
    a=b;
    b=c;
  }
  for (d=1; (a & 1) == 0 && (b & 1) == 0; d<<=1) {
    a>>=1;
  }
  b>>=1;
  c=(a & 1) != 0 ? -b : a;
  while (c != 0) {
    while ((c & 1) == 0) {
      c>>=1;
    }
    if (c > 0) {
      a=c;
    }
 else {
      b=-c;
    }
    c=a - b;
  }

}

private static void fn_A(char[] a,int b,char[] c){
  boolean d=b == 0;
  if (d || '9' < a[b - 1] || a[b - 1] < '0') {
    c[b]='1';
  }
 else {
    char e=a[b - 1];
    boolean f=b - 2 >= 0;
    boolean g=f && '0' <= a[b - 2] && a[b - 2] <= '9';
    int h=g ? (a[b - 2] - '0') * 10 + e - '0' : e - '0';
    h++;
    int i=g ? b - 2 : b - 1;
    if (h > 9) {
      c[i++]=(char)('0' + h / 10);
      c[i++]=(char)('0' + h - h / 10 * 10);
    }
 else {
      c[i++]=(char)('0' + h);
    }
  }
}

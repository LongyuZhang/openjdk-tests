static int fn_A(int a){
  if (a == 0) {
    return 32;
  }
  int b=0;
  if (a >>> 16 != 0) {
    a>>>=16;
    b|=16;
  }
  if (a >>> 8 != 0) {
    a>>>=8;
    b|=8;
  }
  if (a >>> 4 != 0) {
    a>>>=4;
    b|=4;
  }
  if (a >>> 2 != 0) {
    a>>>=2;
    b|=2;
  }
  if (a >>> 1 != 0) {
    a>>>=1;
    b|=1;
  }
  return 32 - b - 1;
}

private static void fn_A(int a,int[] b,int c,int[] d,int e){
  d[0 + e]=(b[1 + c] << (a - 32)) | (b[2 + c] >>> (64 - a));
  d[1 + e]=(b[2 + c] << (a - 32)) | (b[3 + c] >>> (64 - a));
  d[2 + e]=(b[3 + c] << (a - 32)) | (b[0 + c] >>> (64 - a));
  d[3 + e]=(b[0 + c] << (a - 32)) | (b[1 + c] >>> (64 - a));
  b[0 + c]=d[0 + e];
  b[1 + c]=d[1 + e];
  b[2 + c]=d[2 + e];
  b[3 + c]=d[3 + e];
}

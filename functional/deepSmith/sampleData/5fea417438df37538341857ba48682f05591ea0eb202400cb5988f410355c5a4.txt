public static String fn_A(String[] a,char b){
  for (int c=0; c < a.length; c++) {
    if (a[c].charAt(0) == '-') {
      if (a[c].charAt(1) == b) {
        if (a[c].length() > 2) {
          String d=a[c].substring(2).trim();
          return d;
        }
 else         if (c < a.length - 1) {
          if (a[c + 1].charAt(0) == '-') {
            return null;
          }
          return a[c + 1];
        }
      }
    }
  }
  return null;
}

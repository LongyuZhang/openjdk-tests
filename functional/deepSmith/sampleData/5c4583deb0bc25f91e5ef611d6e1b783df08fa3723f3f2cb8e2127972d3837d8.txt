public static int fn_A(long a,int b){
  long c=(a + 7) / 8;
  int d=(1 << b) - 1;
  if ((d & c) != 0) {
    c>>=b;
    ++c;
    c<<=b;
  }
  if (c > Integer.MAX_VALUE) {
    throw new IllegalArgumentException("byteSize=" + c + " too "+ "large for bitSize="+ a+ ", foldFactor="+ b);
  }
  return (int)c;
}

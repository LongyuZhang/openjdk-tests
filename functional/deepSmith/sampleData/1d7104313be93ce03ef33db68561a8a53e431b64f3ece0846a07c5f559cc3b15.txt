private static int fn_A(int a){
  if (a == 0)   return 0;
  if (a == 1)   return 5000000;
  if (a >= 2 && a <= 3)   return 3000000;
  if (a >= 4 && a <= 6)   return 2000000;
  if (a >= 7 && a <= 10)   return 500000;
  if (a >= 11 && a <= 15)   return 300000;
  if (a >= 16 && a <= 21)   return 100000;
  return 0;
}

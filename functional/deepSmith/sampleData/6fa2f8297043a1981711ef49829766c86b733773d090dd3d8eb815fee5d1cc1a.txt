public static Class fn_A(Class a){
  if (a == Boolean.TYPE) {
    return Boolean.class;
  }
  if (a == Byte.TYPE) {
    return Byte.class;
  }
  if (a == Short.TYPE) {
    return Short.class;
  }
  if (a == Character.TYPE) {
    return Character.class;
  }
  if (a == Integer.TYPE) {
    return Integer.class;
  }
  if (a == Long.TYPE) {
    return Long.class;
  }
  if (a == Float.TYPE) {
    return Float.class;
  }
  if (a == Double.TYPE) {
    return Double.class;
  }
  return a;
}

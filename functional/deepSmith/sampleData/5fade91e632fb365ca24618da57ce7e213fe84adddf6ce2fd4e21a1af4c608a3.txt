private static int fn_A(int a,int b,double c){
  int d=(a & 0xFF000000) >>> 24;
  int e=(a & 0x00FF0000) >> 16;
  int f=(a & 0x0000FF00) >> 8;
  int g=(a & 0x000000FF);
  int h=(b & 0xFF000000) >>> 24;
  int i=(b & 0x00FF0000) >> 16;
  int j=(b & 0x0000FF00) >> 8;
  int k=(b & 0x000000FF);
  int l=((int)(d * (1.0 - c) + h * c) << 24) | ((int)(e * (1.0 - c) + i * c) << 16) | ((int)(f * (1.0 - c) + j * c) << 8)| (int)(g * (1.0 - c) + k * c);
  return l;
}

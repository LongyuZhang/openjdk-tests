public static boolean fn_A(double a,double b,int c){
  long d=Double.doubleToLongBits(a);
  if (d < 0) {
    d=0x8000000000000000L - d;
  }
  long e=Double.doubleToLongBits(b);
  if (e < 0) {
    e=0x8000000000000000L - e;
  }
  long f=Math.abs(d - e);
  if (f <= c) {
    return true;
  }
  return false;
}

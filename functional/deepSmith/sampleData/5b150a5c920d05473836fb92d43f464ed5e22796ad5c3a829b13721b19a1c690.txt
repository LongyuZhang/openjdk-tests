public static double[][] fn_A(double a[][]){
  int b=a.length;
  int c=a[0].length;
  double d[][];
  if (b == 1) {
    d=(double[][])a.clone();
  }
 else {
    d=new double[1][c];
    int e=0;
    for (int f=0; f < b; f++) {
      for (int g=0; g < c; g++)       d[0][e++]=a[f][g];
    }
  }
  return d;
}

private static int fn_A(float[] a,float[] b,float[] c){
  int d=0;
  int e=0;
  int f=0;
  while (d < a.length && e < b.length) {
    final float g=a[d];
    final float h=b[e];
    if (g <= h) {
      if (c != null) {
        c[f]=g;
      }
      d++;
    }
    if (h <= g) {
      if (c != null) {
        c[f]=h;
      }
      e++;
    }
    f++;
  }
  return f;
}

public static byte[] fn_A(float[] a,int b,int c){
  byte[] d=new byte[4 * c];
  for (int e=b; e < (b + c); e++) {
    int f=Float.floatToIntBits(a[e]);
    d[4 * (e - b)]=(byte)f;
    d[4 * (e - b) + 1]=(byte)(f >> 8);
    d[4 * (e - b) + 2]=(byte)(f >> 16);
    d[4 * (e - b) + 3]=(byte)(f >> 24);
  }
  return d;
}

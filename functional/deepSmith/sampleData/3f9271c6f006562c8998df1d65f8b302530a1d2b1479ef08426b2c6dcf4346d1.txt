public static double[][] fn_A(double[][] a,int[] b){
  double[][] c=new double[b.length][a[0].length];
  for (int d=0; d < b.length; d++) {
    for (int e=0; e < a[0].length; e++) {
      if ((b[d] < 0) || (b[d] >= a.length))       throw new IllegalArgumentException("Argument out of range.");
      c[d][e]=a[b[d]][e];
    }
  }
  return c;
}

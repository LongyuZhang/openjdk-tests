private static int fn_A(int a,int b){
  if (a >= 1536) {
    return b <= 100 ? 3 : b <= 128 ? 4 : 4 + (b - 128 + 1) / 2;
  }
 else   if (a >= 1024) {
    return b <= 100 ? 4 : b <= 112 ? 5 : 5 + (b - 112 + 1) / 2;
  }
 else   if (a >= 512) {
    return b <= 80 ? 5 : b <= 100 ? 7 : 7 + (b - 100 + 1) / 2;
  }
 else {
    return b <= 80 ? 40 : 40 + (b - 80 + 1) / 2;
  }
}

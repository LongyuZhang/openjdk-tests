private static long fn_A(int[] a,int b,int c){
  int d=c >> 5;
  c&=31;
  int e=a[b - 1] < 0 ? -1 : 0;
  int f=d >= b ? e : a[d];
  d++;
  int g=d >= b ? e : a[d];
  if (c != 0) {
    d++;
    int h=d >= b ? e : a[d];
    f=(f >>> c) | (g << (32 - c));
    g=(g >>> c) | (h << (32 - c));
  }
  return ((long)g << 32) | ((long)f & 0xffffffffL);
}

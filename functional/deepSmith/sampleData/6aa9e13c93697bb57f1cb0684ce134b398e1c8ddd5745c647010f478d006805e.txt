public final static String fn_A(long a){
  long b=60 * 60 * 1000;
  long c=b * 24;
  long d=c * 30;
  long e=365 * c;
  long f=System.currentTimeMillis() - a;
  System.out.println(System.currentTimeMillis() + "differnt:  " + f);
  if (f < b) {
    return "";
  }
 else   if (f < c) {
    return f / b + "";
  }
 else   if (f < d) {
    return f / c + "";
  }
 else   if (f < e) {
    return f / d + "";
  }
 else {
    return f / e + "";
  }
}

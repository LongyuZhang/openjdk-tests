public static void fn_A(int a,int b,float[] c,float[] d,float[] e,int[] f){
  int g=c.length / b;
  for (int h=0; h < g; h++) {
    int i=-1;
    float j=Float.MAX_VALUE;
    for (int k=0; k < a; k++) {
      float l=0;
      for (int m=0; m < b; m++) {
        float n=c[h * b + m] - d[k * b + m];
        l+=n * n;
      }
      if (l < j) {
        j=l;
        i=k;
      }
    }
    for (int m=0; m < b; m++) {
      e[i * b + m]+=c[h * b + m];
    }
    f[i]++;
  }
}

public static int fn_A(int a){
  int b=a;
  int c=0;
  int d=0;
  while ((b & 1) == 1) {
    d++;
    b>>=1;
  }
  if (b == 0) {
    return -1;
  }
  while (((b & 1) == 0) && (b != 0)) {
    c++;
    b>>=1;
  }
  int e=c + d;
  a&=((~0) << (e + 1));
  int f=(1 << (d + 1)) - 1;
  a|=f << (c - 1);
  return a;
}

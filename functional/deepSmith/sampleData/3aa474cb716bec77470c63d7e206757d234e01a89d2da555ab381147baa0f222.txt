public static void fn_A(long a){
  long b, c;
  b=c=System.currentTimeMillis();
  do {
    try {
      Thread.sleep(a - (c - b));
    }
 catch (    InterruptedException d) {
      System.out.println("Test caught interrupted exception");
    }
    c=System.currentTimeMillis();
  }
 while ((c - b) < a);
}

public static void fn_A(int[][] a){
  boolean[] b=new boolean[a.length];
  boolean[] c=new boolean[a[0].length];
  for (int d=0; d < a.length; d++) {
    for (int e=0; e < a[0].length; e++) {
      if (a[d][e] == 0) {
        b[d]=true;
        c[e]=true;
      }
    }
  }
  for (int d=0; d < b.length; d++) {
    for (int e=0; e < c.length; e++) {
      if (b[d] || c[e])       a[d][e]=0;
    }
  }
}

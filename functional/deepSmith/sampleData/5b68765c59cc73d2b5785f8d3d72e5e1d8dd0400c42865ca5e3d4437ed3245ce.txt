public static long[] fn_A(int[] a){
  long[] b=new long[3];
  if (a == null || a.length == 0) {
    return b;
  }
  int c=0, d=Integer.MAX_VALUE, e=Integer.MIN_VALUE;
  for (int f=0; f < a.length; f++) {
    c+=a[f];
    if (a[f] > e) {
      e=a[f];
    }
    if (a[f] < d) {
      d=a[f];
    }
  }
  b[0]=c;
  b[1]=d;
  b[2]=e;
  return b;
}

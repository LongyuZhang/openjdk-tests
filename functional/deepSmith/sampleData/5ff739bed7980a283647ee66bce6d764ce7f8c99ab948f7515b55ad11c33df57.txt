public static int fn_A(int a[],int b,int c){
  double d=b;
  double e=0;
  for (int f=0; f < b; f++)   e+=(f / d) * a[f];
  double g=0;
  int h=0;
  double i=0;
  int j=0;
  for (int f=0; f < b; f++) {
    h+=a[f];
    if (h == 0)     continue;
    int k=c - h;
    if (k == 0)     break;
    g+=(f / d) * a[f];
    double l=g / h;
    double m=(e - g) / k;
    double n=(double)h * (double)k * (l - m)* (l - m);
    if (n > i) {
      i=n;
      j=f;
    }
  }
  return j;
}

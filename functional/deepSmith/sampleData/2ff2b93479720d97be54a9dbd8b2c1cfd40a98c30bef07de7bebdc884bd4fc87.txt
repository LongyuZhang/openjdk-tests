public static BigDecimal fn_A(Object a){
  if (a instanceof Double)   return new BigDecimal((Double)a);
 else   if (a instanceof Float)   return new BigDecimal((Float)a);
 else   if (a instanceof BigDecimal)   return ((BigDecimal)a);
 else   if (a instanceof Long)   return new BigDecimal((Long)a);
 else   if (a instanceof Integer)   return new BigDecimal((Integer)a);
 else   if (a instanceof Short)   return new BigDecimal((Short)a);
 else   if (a instanceof Byte)   return new BigDecimal((Byte)a);
 else {
    throw null;
  }
}

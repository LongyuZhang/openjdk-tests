protected static void fn_A(int a,int b,float[] c){
  int d, e;
  float f;
  int g=a / 2;
  int h=b / 2;
  for (int i=0; i < h; i++) {
    e=i + h;
    for (int j=0; j < a; j++) {
      f=c[j + a * i];
      c[j + a * i]=c[j + a * e];
      c[j + a * e]=f;
    }
  }
  for (int j=0; j < g; j++) {
    d=j + g;
    for (int i=0; i < b; i++) {
      f=c[j + a * i];
      c[j + a * i]=c[d + a * i];
      c[d + a * i]=f;
    }
  }
}

private static int fn_A(int[] a,int[] b,short[] c,int d,int e){
  int f=d - 1;
  int g=e;
  while (true) {
    while (b[++f] < b[e])     ;
    while (b[e] < b[--g])     if (g == d)     break;
    if (f >= g)     break;
    int h=b[f];
    b[f]=b[g];
    b[g]=h;
    int i=a[f];
    a[f]=a[g];
    a[g]=i;
    short j=c[f];
    c[f]=c[g];
    c[g]=j;
  }
  int h=b[f];
  b[f]=b[e];
  b[e]=h;
  int i=a[f];
  a[f]=a[e];
  a[e]=i;
  short j=c[f];
  c[f]=c[e];
  c[e]=j;
  return f;
}

public static void fn_A(final long a,final long b,final long c){
  if (b < 0)   throw new ArrayIndexOutOfBoundsException("Offset (" + b + ") is negative");
  if (c < 0)   throw new IllegalArgumentException("Length (" + c + ") is negative");
  if (b + c > a)   throw new ArrayIndexOutOfBoundsException("Last index (" + (b + c) + ") is greater than big-array length ("+ a+ ")");
}

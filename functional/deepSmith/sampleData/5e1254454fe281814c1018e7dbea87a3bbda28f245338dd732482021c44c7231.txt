public static void fn_A(float[][] a){
  int b=a[0].length;
  int c=a.length;
  for (int d=0; d < c; d++) {
    float[] e=(float[])a[d].clone();
    float[] f=(float[])a[((d - 1) + c) % c].clone();
    float[] g=(float[])a[(d + 1) % c].clone();
    for (int h=0; h < b; h++) {
      a[d][h]=((((((e[h] + f[h]) + g[h]) * 2.0f) + ((e[(h + 1) % b] + e[((h - 1) + b) % b]) * 2.0f)) + (f[(h + 1) % b] + f[((h - 1) + b) % b])) + (g[(h + 1) % b] + g[((h - 1) + b) % b])) / 14.0f;
    }
  }
}

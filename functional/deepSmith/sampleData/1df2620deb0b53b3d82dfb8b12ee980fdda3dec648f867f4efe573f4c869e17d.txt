public static double fn_A(int a){
  if (a == 1)   return 0.08;
  if (a == 2)   return 0.125;
  if (a >= 3 && a <= 5)   return 0.167;
  if (a >= 6 && a <= 8)   return 0.208;
  if (a >= 9 && a <= 11)   return 0.25;
  if (a >= 12 && a <= 14)   return 0.33;
  if (a >= 15 && a <= 18)   return 0.41;
  return 0.5;
}

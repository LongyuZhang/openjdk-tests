public static void main(String[] a){
  double b=0.0;
  for (int c=50000; c >= 1; c--) {
    b+=1.0 / c;
  }
  System.out.println("Computing from right to left " + b);
  double d=0.0;
  for (int c=1; c <= 50000; c++) {
    d+=1.0 / c;
  }
  System.out.println("Computing from left to right " + d);
  double e=b - d;
  System.out.println("difference is " + e);
}

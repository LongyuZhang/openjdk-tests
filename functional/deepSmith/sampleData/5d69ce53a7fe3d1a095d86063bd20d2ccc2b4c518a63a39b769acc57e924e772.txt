private static double[] fn_A(double[][] a,double[] b){
  int c=a[0].length;
  for (int d=0; d < c - 1; ++d) {
    for (int e=d + 1; e < c; ++e) {
      double f=a[e][d] / a[d][d];
      for (int g=d + 1; g < c; ++g) {
        a[e][g]-=f * a[d][g];
      }
      b[e]-=f * b[d];
    }
  }
  double[] h=new double[c];
  for (int d=c - 1; d >= 0; --d) {
    double i=b[d];
    for (int g=d + 1; g < c; ++g) {
      i-=a[d][g] * h[g];
    }
    h[d]=i / a[d][d];
  }
  return h;
}

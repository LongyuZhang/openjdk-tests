protected static char fn_A(int[] a){
  int b=a[0];
  int c=a[1];
  int d=a[2];
  int e=a[3];
  if (b >= c && b >= d && b >= e) {
    return 'A';
  }
  if (c >= b && c >= d && c >= e) {
    return 'C';
  }
  if (d >= b && d >= c && d >= e) {
    return 'G';
  }
  if (e >= b && e >= c && e >= d) {
    return 'T';
  }
  return '?';
}

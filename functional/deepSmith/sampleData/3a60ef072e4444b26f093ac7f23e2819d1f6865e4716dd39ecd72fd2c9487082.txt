public static boolean fn_A(Object a){
  if (a == null) {
    return false;
  }
 else   if (a instanceof Boolean) {
    return (Boolean)a;
  }
 else   if (a instanceof String) {
    return !((String)a).isEmpty();
  }
 else   if (a instanceof Collection) {
    return !((Collection<?>)a).isEmpty();
  }
 else   if (a instanceof Map) {
    return !((Map<?,?>)a).isEmpty();
  }
 else   if (a instanceof Iterator) {
    return ((Iterator<?>)a).hasNext();
  }
 else   if (a instanceof Enumeration) {
    return ((Enumeration<?>)a).hasMoreElements();
  }
 else {
    return true;
  }
}

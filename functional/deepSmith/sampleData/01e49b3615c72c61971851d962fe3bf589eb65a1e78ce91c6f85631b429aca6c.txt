public static int[][] fn_A(int[] a){
  int b=a.length;
  int[][] c=new int[b][b];
  for (int d=0; d < b; ++d) {
    c[d][d]=d;
  }
  for (int d=0; d < b; ++d) {
    for (int e=d + 1; e < b; ++e) {
      if (a[c[d][e - 1]] > a[e]) {
        c[d][e]=e;
      }
 else {
        c[d][e]=c[d][e - 1];
      }
    }
  }
  return c;
}

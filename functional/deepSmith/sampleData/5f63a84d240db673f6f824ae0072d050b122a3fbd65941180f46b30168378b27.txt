public static int[] fn_A(int[] a,int[] b){
  int[] c=new int[a.length + b.length];
  for (int d=0; d < a.length; d++)   c[d]=a[d];
  int e=a.length;
  for (int d=0; d < b.length; d++) {
    int f=0;
    for (int g=0; g < a.length; g++) {
      if (b[d] == c[g])       f=1;
    }
    if (f == 0) {
      c[e]=b[d];
      e++;
    }
  }
  return c;
}

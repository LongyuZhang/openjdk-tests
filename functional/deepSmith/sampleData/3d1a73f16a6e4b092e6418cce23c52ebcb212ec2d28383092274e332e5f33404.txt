private static float fn_A(float a,float b){
  if (a > 0) {
    return (float)Math.atan((double)b / a);
  }
 else   if ((a < 0) && (b >= 0)) {
    return (float)(Math.atan((double)b / a) + Math.PI);
  }
 else   if ((a < 0) && (b < 0)) {
    return (float)(Math.atan((double)b / a) - Math.PI);
  }
 else   if ((a == 0) && (b > 0)) {
    return (float)Math.PI / 2;
  }
 else   if ((a == 0) && (b < 0)) {
    return (float)-Math.PI / 2;
  }
 else {
    return (float)0;
  }
}

public static double fn_A(double[] a,double[] b){
  double c=0;
  double d=0;
  double e;
  if (a.length < b.length) {
    e=a.length;
  }
 else {
    e=b.length;
  }
  if (e < 1) {
    throw new RuntimeException("Array is to small for calculation error");
  }
  for (int f=1; f < e; f++) {
    c+=Math.pow(a[f - 1] - b[f - 1],2);
    d+=Math.pow(a[f - 1] - a[f],2);
  }
  return c / d;
}

public static void fn_A(int[][] a,int b){
  for (int c=0; c < b / 2; c++) {
    int d=b - 2 * c - 1;
    int e=b - 1 - c;
    for (int f=0; f < d; f++) {
      int g=a[c][c + f];
      a[c][c + f]=a[e - f][c];
      a[e - f][c]=a[e][e - f];
      a[e][e - f]=a[c + f][e];
      a[c + f][e]=g;
    }
  }
}

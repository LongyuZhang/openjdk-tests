public static void fn_A(double[] a,int b,int c,int d[]){
  for (int e=0; e < c; e++) {
    d[e]=b + e;
  }
  int e, f;
  int g=1;
  double h;
  do {
    g*=3;
    g++;
  }
 while (g <= c);
  do {
    g/=3;
    for (e=g; e < c; e++) {
      h=a[d[e]];
      int i=d[e];
      f=e;
      while (a[d[f - g]] > h) {
        d[f]=d[f - g];
        f-=g;
        if (f < g)         break;
      }
      d[f]=i;
    }
  }
 while (g > 1);
}

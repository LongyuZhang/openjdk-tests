static int fn_A(int[] a,int b,int c){
  int d=0;
  for (int e=0; e < b; e++) {
    if (a[e] < 0) {
      d+=3;
    }
 else     if (a[e] == 0) {
      if (c > 0) {
        c-=1;
        d+=3;
      }
 else {
        d+=1;
      }
    }
 else     if (a[e] > 0) {
      if (c >= a[e] + 1) {
        c-=a[e] + 1;
        d+=3;
      }
 else       if (c >= a[e]) {
        c-=a[e];
        d+=1;
      }
    }
  }
  return d;
}

private static int fn_A(int[] a,int[] b,int c,int d){
  int e=0, f=0, g=0;
  while (true) {
    final int h=a[e];
    final int i=b[f];
    if (h <= i) {
      e++;
      g++;
      if (e >= c) {
        if (h == i) {
          f++;
        }
        g+=d - f;
        break;
      }
      if (h == i) {
        f++;
      }
 else {
        continue;
      }
    }
 else {
      f++;
      g++;
    }
    if (f >= d) {
      g+=c - e;
      break;
    }
  }
  return g;
}

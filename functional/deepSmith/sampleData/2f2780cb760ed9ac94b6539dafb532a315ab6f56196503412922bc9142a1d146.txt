public static int[][] fn_A(char[] a,char[] b){
  int[][] c=new int[a.length][b.length];
  c[0][0]=a[0] == b[0] ? 1 : 0;
  for (int d=1; d < a.length; d++) {
    if (a[d] == b[0]) {
      c[d][0]=1;
    }
  }
  for (int e=1; e < b.length; e++) {
    if (a[0] == b[e]) {
      c[0][e]=1;
    }
  }
  for (int d=1; d < a.length; d++) {
    for (int e=1; e < b.length; e++) {
      if (a[d] == b[e]) {
        c[d][e]=c[d - 1][e - 1] + 1;
      }
    }
  }
  return c;
}

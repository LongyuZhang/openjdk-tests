public static double[] fn_A(double[] a,double[] b,double[] c,double[] d){
  double e=c[0] * d[0] + c[1] * d[1] + c[2] * d[2];
  double f=a[0];
  double g=a[1];
  double h=a[2];
  double i=b[0];
  double j=b[1];
  double k=b[2];
  double l=d[0];
  double m=d[1];
  double n=d[2];
  double o=-(l * f + m * g + n * h - e) / (l * i + m * j + n * k);
  return new double[]{f + o * i,g + o * j,h + o * k};
}

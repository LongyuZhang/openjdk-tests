public static int fn_A(int a){
  if (a <= 170)   return 4;
  if (a >= 171 && a < 180)   return 5;
  if (a >= 180 && a < 200)   return 6;
  if (a >= 200 && a < 210)   return 7;
  if (a >= 210 && a < 220)   return 8;
  if (a >= 220 && a < 230)   return 10;
  if (a >= 230 && a <= 240)   return 12;
  return 10;
}

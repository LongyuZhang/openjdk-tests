private static int[][] fn_A(int[][] a){
  int[][] b=new int[a.length][a.length];
  int c=a.length - 1;
  b[0][c]=a[0][c];
  for (int d=c - 1; d >= 0; d--) {
    b[0][d]=b[0][d + 1] + a[0][d];
    int e=c - d;
    b[e][c]=b[e - 1][c] + a[e][c];
  }
  for (int e=1; e < a.length; e++) {
    for (int f=a.length - 2; f >= 0; f--) {
      b[e][f]=Math.max(b[e - 1][f],b[e][f + 1]) + a[e][f];
    }
  }
  return b;
}

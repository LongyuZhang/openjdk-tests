private static int fn_A(int a) throws NumberFormatException {
  if (a >= 0x30 && a < 0x3A)   return a - 0x30;
 else   if (a >= 0x41 && a < 0x47)   return a - 0x37;
 else   if (a >= 0x61 && a < 0x67)   return a - 0x57;
 else   throw new NumberFormatException('\'' + a + "' is not a valid hexadecimal digit.");
}

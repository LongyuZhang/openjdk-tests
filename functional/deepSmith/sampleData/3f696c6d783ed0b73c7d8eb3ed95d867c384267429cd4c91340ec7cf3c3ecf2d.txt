public static int fn_A(long[] a,long[] b,int c,int d){
  final long[] e=new long[2];
  while (c <= d) {
    final int f=(c + d) / 2;
    e[0]=a[f * 2];
    e[1]=a[f * 2 + 1];
    if ((e[0] < b[0]) || ((e[0] == b[0]) && (e[1] < b[1]))) {
      c=f + 1;
    }
 else     if ((e[0] > b[0]) || ((e[0] == b[0]) && (e[1] > b[1]))) {
      d=f - 1;
    }
 else {
      return f;
    }
  }
  return -(c + 1);
}

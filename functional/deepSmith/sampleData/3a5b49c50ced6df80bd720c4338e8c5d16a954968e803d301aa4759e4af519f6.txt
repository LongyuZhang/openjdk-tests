public static Float fn_A(Object a){
  if (a instanceof Double)   return ((Double)a).floatValue();
 else   if (a instanceof Float)   return ((Float)a);
 else   if (a instanceof BigDecimal)   return ((BigDecimal)a).floatValue();
 else   if (a instanceof Long)   return ((Long)a).floatValue();
 else   if (a instanceof Integer)   return ((Integer)a).floatValue();
 else   if (a instanceof Short)   return ((Short)a).floatValue();
 else   if (a instanceof Byte)   return ((Byte)a).floatValue();
 else {
    return null;
  }
}

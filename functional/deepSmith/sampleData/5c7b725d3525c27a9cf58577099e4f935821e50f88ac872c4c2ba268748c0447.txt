private static String fn_A(long a,long b){
  if (b <= 0) {
    return "0 B/s";
  }
  if (a <= 0) {
    return "0 B/s";
  }
  float c=((float)a) * 1000.f / b;
  if (c >= 1000 * 1000) {
    return String.format(Locale.US,"%.2f MB/s",((float)c) / 1000 / 1000);
  }
 else   if (c >= 1000) {
    return String.format(Locale.US,"%.1f KB/s",((float)c) / 1000);
  }
 else {
    return String.format(Locale.US,"%d B/s",(long)c);
  }
}

public static double fn_A(float[] a,int b,int c,int d){
  double e=(a[c] - a[b]);
  double f=(a[c + 1] - a[b + 1]);
  double g=Math.sqrt(e * e + f * f);
  double h=(a[c] - a[d]);
  double i=(a[c + 1] - a[d + 1]);
  double j=Math.sqrt(h * h + i * i);
  double k=g * j;
  if (k <= 0)   return 0;
  double l=(e * -h + f * -i) / k;
  if (l > 1)   l=1;
 else   if (l < -1)   l=-1;
  return l;
}

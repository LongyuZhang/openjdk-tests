protected static double[][][] fn_A(int a,int b) throws IllegalArgumentException {
  if (a >= 0x01 && b >= 0x01) {
    double[][][] c=new double[b][a][b];
    double d=1.0d / b;
    for (int e=0x00; e < b; e++) {
      for (int f=0x00; f < a; f++) {
        for (int g=0x00; g < b; g++) {
          c[e][f][g]=d;
        }
      }
    }
    return c;
  }
 else {
    throw new IllegalArgumentException("The number of states and symbols must both be larger or equal to one.");
  }
}

static void fn_A(float[] a,float[] b){
  float c, d, e, f, g, h, i, j, k;
  c=(b[2] * 0.7071067811865476f);
  d=b[0] - c;
  e=b[0] + c;
  f=b[1] + b[3];
  g=(b[1] * 1.3065629648763766f);
  h=(f * (-0.9238795325112866f));
  i=(b[3] * (-0.5411961001461967f));
  j=g + h;
  k=i - h;
  a[3]=e - k;
  a[0]=e + k;
  a[2]=d - j;
  a[1]=d + j;
}

private static <T extends Comparable<? super T>>void fn_A(T[] a,T[] b,int c,int d,int e){
  for (int f=c; f <= e; f++) {
    b[f]=a[f];
  }
  int f=c;
  int g=d + 1;
  int h=c;
  while (f <= d && g <= e) {
    if (b[f].compareTo(b[g]) <= 0) {
      a[h]=b[f];
      f++;
    }
 else {
      a[h]=b[g];
      g++;
    }
    h++;
  }
  while (f <= d) {
    a[h]=b[f];
    h++;
    f++;
  }
}

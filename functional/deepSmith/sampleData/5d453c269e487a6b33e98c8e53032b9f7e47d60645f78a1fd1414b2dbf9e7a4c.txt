static double fn_A(final double a){
  if (a == Double.POSITIVE_INFINITY) {
    return 3 * Math.PI / 4;
  }
 else   if (a == Double.NEGATIVE_INFINITY) {
    return -3 * Math.PI / 4;
  }
 else   if (a > 0.0) {
    return Math.PI;
  }
 else   if (a < 0.0) {
    return -Math.PI;
  }
 else {
    return Double.NaN;
  }
}

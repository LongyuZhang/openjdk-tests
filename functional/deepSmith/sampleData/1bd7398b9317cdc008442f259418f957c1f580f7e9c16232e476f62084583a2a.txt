public static double fn_A(double a,double b){
  if (a >= 0. && b > 0.) {
    return Math.atan(b / a);
  }
 else   if (a <= 0. && b > 0.) {
    return Math.PI - Math.atan(b / -a);
  }
 else   if (a <= 0. && b < 0.) {
    return Math.PI + Math.atan(b / a);
  }
 else   if (a >= 0. && b < 0.) {
    return Math.PI * 2. - Math.atan(-b / a);
  }
 else   if (b == 0. && a != 0.) {
    return (a < 0.) ? Math.PI : 0.;
  }
 else {
    return 0.;
  }
}

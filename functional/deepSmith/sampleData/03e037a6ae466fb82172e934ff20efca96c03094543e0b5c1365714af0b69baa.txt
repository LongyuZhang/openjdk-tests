public static void main(String[] a){
  double b=0.0;
  for (double c=1.0; c <= 50000.0; c++) {
    b+=1 / c;
  }
  System.out.println("Summation of series left to right: " + b);
  double d=0.0;
  for (double c=50000.0; c >= 1.0; c--) {
    d+=1 / c;
  }
  System.out.println("Summation of series right to left: " + d);
  System.out.println("Summation of the series right to left - " + "Summation of the series left to right: " + (d - b));
}

protected static int fn_A(char a){
  if (a >= 'A' && a <= 'Z') {
    return a - 65;
  }
  if (a >= 'a' && a <= 'z') {
    return (a - 97) + 26;
  }
  if (a >= '0' && a <= '9') {
    return (a - 48) + 52;
  }
  if (a == '+') {
    return 62;
  }
  if (a == '/') {
    return 63;
  }
  return a != '=' ? -1 : 0;
}

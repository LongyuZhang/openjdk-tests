public static final String fn_A(byte a){
switch (a) {
case 1:
    return "WELCOME";
case 2:
  return "READY";
case 3:
return "FIND_COORD";
case 4:
return "SET_MANIFEST";
case 5:
return "RELAY";
case 6:
return "CREATE_FEDERATION";
case 7:
return "JOIN_FEDERATION";
case 8:
return "RESIGN_FEDERATION";
case 9:
return "DESTROY_FEDERATION";
case 127:
return "BUNDLE";
default :
return "UNKNOWN";
}
}

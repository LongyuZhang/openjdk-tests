public static int fn_A(int[] a,int b){
  int c=0;
  int d=a.length - 1;
  int e=-1;
  if (b < a[c] || b > a[d] || c > d) {
    return -1;
  }
  while (c <= d) {
    e=(c + d) >>> 1;
    if (b < a[e]) {
      d=e - 1;
    }
 else     if (b > a[e]) {
      c=e + 1;
    }
 else {
      return e;
    }
  }
  return -1;
}

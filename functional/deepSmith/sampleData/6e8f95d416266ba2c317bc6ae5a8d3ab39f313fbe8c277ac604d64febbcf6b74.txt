public static boolean fn_A(Object a){
  if (a == null)   return false;
 else   if (a instanceof Object[])   return true;
 else   if (a instanceof char[])   return true;
 else   if (a instanceof byte[])   return true;
 else   if (a instanceof short[])   return true;
 else   if (a instanceof int[])   return true;
 else   if (a instanceof long[])   return true;
 else   if (a instanceof float[])   return true;
 else   if (a instanceof double[])   return true;
 else   if (a instanceof boolean[])   return true;
 else   return false;
}

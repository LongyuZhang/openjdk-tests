public static float[] fn_A(float[] a,int b,int c,boolean d){
  float e=a[0], f=a[0];
  int g=0;
  for (int h=0; h < c; h++) {
    for (int i=0; i < b; i++) {
      if (d) {
        g=2 * (i + h * b);
      }
 else {
        g=i + h * b;
      }
      if (a[g] < e) {
        e=a[g];
      }
      if (a[g] > f) {
        f=a[g];
      }
    }
  }
  return new float[]{e,f};
}

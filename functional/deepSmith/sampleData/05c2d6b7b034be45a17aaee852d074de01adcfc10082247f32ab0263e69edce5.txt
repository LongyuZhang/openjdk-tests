private static boolean fn_A(double a,double b,double c,double d,double e,double f,double g,double h){
  double i, j, k, l, m, n, o, p, q, r, s, t;
  double u, v, w;
  i=e - c;
  j=f - d;
  k=a - e;
  l=b - f;
  m=c - a;
  n=d - b;
  o=g - a;
  p=h - b;
  q=g - c;
  r=h - d;
  s=g - e;
  t=h - f;
  w=i * r - j * q;
  u=m * p - n * o;
  v=k * t - l * s;
  return ((w >= 0.0f) && (v >= 0.0f) && (u >= 0.0f));
}

private static int fn_A(int[] a){
  int b=0;
  int c=0;
  int d=0;
  int e=0;
  for (int f=0; f < a.length; f++) {
    c=c + a[f];
    if (c < 0) {
      c=0;
      e=f + 1;
    }
 else     if (c > b) {
      b=c;
      d=f;
    }
  }
  System.out.println(d + "" + a[d]);
  System.out.println(e + "->" + a[e]);
  return b;
}

public static final byte fn_A(int a){
  if (a <= 0) {
    return -1;
  }
  byte b=0;
  if (a >= 1 << 16) {
    a>>=16;
    b+=16;
  }
  if (a >= 1 << 8) {
    a>>=8;
    b+=8;
  }
  if (a >= 1 << 4) {
    a>>=4;
    b+=4;
  }
  if (a >= 1 << 2) {
    a>>=2;
    b+=2;
  }
  if (a >= 1 << 1) {
    a>>=1;
    b+=1;
  }
  return b;
}

public static int[] fn_A(int[] a){
  int b=0;
  for (int c=0; c < a.length; ++c) {
    if (a[c] != Integer.MIN_VALUE) {
      ++b;
      for (int d=c + 1; d < a.length; ++d) {
        if (a[c] == a[d])         a[d]=Integer.MIN_VALUE;
      }
    }
  }
  int[] e=new int[b];
  int f=0;
  for (int c=0; c < a.length; ++c) {
    if (a[c] != Integer.MIN_VALUE)     e[f++]=a[c];
  }
  return e;
}

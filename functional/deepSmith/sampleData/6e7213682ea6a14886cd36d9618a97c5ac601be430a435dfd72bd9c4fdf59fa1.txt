public static boolean fn_A(final byte[] a,final byte[] b){
  if (a == b) {
    return true;
  }
  int c=a.length;
  if (c != b.length) {
    return false;
  }
  if (0 == c) {
    return true;
  }
  c--;
  if (a[c] == b[c]) {
    for (int d=0; d < c; d++) {
      if (a[d] != b[d]) {
        return false;
      }
    }
  }
 else {
    return false;
  }
  return true;
}

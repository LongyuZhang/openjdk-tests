public static boolean fn_A(int a){
  if (a == 0 || (a > 0 && a < 10))   return true;
  if (a < 0 || (a % 10 == 0 && a != 0))   return false;
  int b=0, c=a;
  while (c != 0) {
    int d=c % 10;
    b=b * 10 + d;
    c/=10;
    System.out.println("num :" + c + " result :"+ b);
  }
  return b == a ? true : false;
}

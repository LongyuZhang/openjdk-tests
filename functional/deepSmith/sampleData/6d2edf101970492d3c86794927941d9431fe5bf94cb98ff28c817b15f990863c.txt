private static String fn_A(int a){
  if ((a >> 30) != 0)   throw new IllegalArgumentException("code is invalid");
  char[] b=new char[6];
  for (int c=5; c >= 0; c--) {
    char d=(char)('A' - 1 + (a & 0x1F));
    if (d < 'A' || d > 'Z')     throw new IllegalArgumentException("code contains invalid character");
    b[c]=d;
    a>>=5;
  }
  return new String(b);
}

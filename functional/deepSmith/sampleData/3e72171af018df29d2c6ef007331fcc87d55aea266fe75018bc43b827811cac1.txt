private static double[] fn_A(double[] a,int b){
  b%=4;
  if (b < 0)   b=4 - b;
  double c[]=new double[]{a[0],a[1],a[2]};
switch (b) {
case 1:
    c[0]=a[2];
  c[2]=1 - a[0];
break;
case 2:
c[0]=1 - a[0];
c[2]=1 - a[2];
break;
case 3:
c[0]=1 - a[2];
c[2]=a[0];
break;
default :
case 0:
break;
}
return c;
}

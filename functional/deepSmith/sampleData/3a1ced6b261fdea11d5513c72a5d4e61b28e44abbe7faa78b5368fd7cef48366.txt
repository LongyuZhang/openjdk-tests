private final static int fn_A(char a){
  if ((a >= '0' && a <= '4')) {
    return 0;
  }
 else   if ((a >= '5' && a <= '9')) {
    return 1;
  }
 else   if (a == 'a' || a == 'b' || a == 'c' || a == 'd' || a == 'e') {
    return 2;
  }
 else   if (a == 'f' || a == 'j' || a == 'l' || a == 'm' || a == 'n') {
    return 3;
  }
 else   if (a == 'z' || a == 'x' || a == 'k' || a == 'w' || a == 'p') {
    return 4;
  }
 else {
    return 5;
  }
}

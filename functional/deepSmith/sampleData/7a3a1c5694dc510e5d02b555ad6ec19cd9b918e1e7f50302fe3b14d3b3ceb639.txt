public static float[][] fn_A(float[] a,int b,int c){
  if (a.length != b * c)   throw new IllegalArgumentException("The size of vector must be the same of product of m and n.");
  int d=0;
  float[][] e=new float[b][c];
  for (int f=0; f < b; f++) {
    for (int g=0; g < c; g++) {
      e[f][g]=a[d++];
    }
  }
  return e;
}

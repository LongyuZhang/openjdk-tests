static void fn_A(int[] a){
  int b=a.length;
  if (b == 0)   System.out.println("" + 0);
  int c=0;
  boolean d=false;
  for (int e=0; e < b; e++) {
    if (a[e] - (e + 1) > 2) {
      d=true;
      break;
    }
    for (int f=Math.max(0,a[e] - 2); f < e; f++)     if (a[f] > a[e])     c++;
  }
  if (!d)   System.out.println("" + c);
 else   System.out.println("Too chaotic");
}

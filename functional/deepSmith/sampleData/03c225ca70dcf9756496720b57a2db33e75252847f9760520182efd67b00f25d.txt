public static int[] fn_A(float[] a,int b){
  if (b < 1) {
    throw new IllegalArgumentException("Maximum int value must be positive. But it is:" + b);
  }
  int[] c=new int[a.length];
  float d=(float)((double)b / 2.0);
  for (int e=0; e < a.length; e++) {
    float f=a[e];
    if (f < -1.0 || f > 1.0) {
      throw new IllegalArgumentException("Array value should be between -1.0 and 1.0. But it is: " + f);
    }
    c[e]=(int)(a[e] * d);
  }
  return c;
}

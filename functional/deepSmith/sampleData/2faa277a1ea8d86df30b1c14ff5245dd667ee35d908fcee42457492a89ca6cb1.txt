public static int[] fn_A(int[][] a){
  final int b=a.length;
  int c=0;
  for (int d=0; d < b; d++)   c+=a[d].length + 1;
  final int[] e=new int[c];
  int f=0;
  for (int d=0; d < b; d++) {
    final int[] g=a[d];
    final int h=g.length;
    e[f++]=h;
    for (int i=0; i < h; i++)     e[f++]=g[i];
  }
  return e;
}

public static String fn_A(int a){
  float b=a;
  float c=1;
  while (b > 750) {
    b/=1000F;
    c*=1000F;
  }
  b=Math.round(b * 100) / 100F;
  if (c == 1000) {
    return b + " KB";
  }
  if (c == 1000 * 1000) {
    return b + " MB";
  }
  if (c == 1000 * 1000 * 1000) {
    return b + " GB";
  }
  return b + " Bytes";
}

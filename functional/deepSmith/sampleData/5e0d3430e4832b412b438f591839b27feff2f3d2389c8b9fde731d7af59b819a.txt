public static void fn_A(Comparable[] a,int b){
  int c=a.length;
  for (int d=0; d < a.length / b; d++) {
    int e=d;
    for (int f=d; f < a.length / b; f++) {
      if (a[f * b].compareTo(a[e * b]) < 0)       e=f;
    }
    for (int g=0; g < b; g++) {
      Comparable h=a[d * b + g];
      a[d * b + g]=a[e * b + g];
      a[e * b + g]=h;
    }
  }
}

public static double fn_A(final double a,final double b,final int c,final int d){
  final int e=d + 2;
  if (c > e) {
    final int f=d * d;
    double g=b;
    double h=g;
    double i=(d + 1) * a * b;
    for (int j=d + 3; j <= c; j++) {
      final int k=j - 1;
      h=g;
      g=i;
      i=k * a * ((2 * j - 3) * g - (j - 2) * h) / (k * k - f);
    }
    return i;
  }
 else   if (c == e) {
    return (d + 1) * a * b;
  }
 else {
    return b;
  }
}

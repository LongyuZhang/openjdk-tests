private static void fn_A(float[] a,int b,int c){
  int d=1;
  int e=1;
  for (int f=c << 1; e < f; e+=2) {
    if (e < d) {
      float g=a[b + e];
      a[b + e]=a[b + d];
      a[b + d]=g;
      float h=a[(b + e) - 1];
      a[(b + e) - 1]=a[(b + d) - 1];
      a[(b + d) - 1]=h;
    }
    int i;
    for (i=c; i > 1 && i < d; i>>=1) {
      d-=i;
    }
    d+=i;
  }
}

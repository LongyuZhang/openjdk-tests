public static void main(String[] a){
  int b=99;
  String c;
  if (b >= 1 && b <= 50) {
    c="Cheap";
  }
 else {
    if (b >= 51 && b <= 100) {
      c="Expensive";
    }
 else {
      c="Not buying";
    }
  }
  System.out.println("Status with If: " + c);
  c=(b >= 1 && b <= 50) ? "Cheap" : (b >= 51 && b <= 100) ? "Expensive" : "Not buying";
  System.out.println("Status with ?: " + c);
}

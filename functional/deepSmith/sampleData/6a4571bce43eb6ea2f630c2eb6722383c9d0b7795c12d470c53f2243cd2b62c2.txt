private static void fn_A(float[] a,int b,byte[] c,int d,int e){
  if (a == null)   throw new NullPointerException();
  if (c == null)   throw new NullPointerException();
  int f=b + e;
  while (b < f) {
    int g=Float.floatToIntBits(a[b++]);
    c[d++]=(byte)((g >> 24) & 0xff);
    c[d++]=(byte)((g >> 16) & 0xff);
    c[d++]=(byte)((g >> 8) & 0xff);
    c[d++]=(byte)((g) & 0xff);
  }
}

private static void fn_A(double[][] a,double[] b){
  for (int c=1; c < a.length; ++c) {
    double d=b[c];
    for (int e=0; e < c; ++e) {
      d-=a[c][e] * b[e];
    }
    b[c]=d;
  }
  b[a.length - 1]/=a[a.length - 1][a.length - 1];
  for (int c=a.length - 2; c >= 0; --c) {
    double d=b[c];
    for (int e=c + 1; e < a.length; ++e) {
      d-=a[c][e] * b[e];
    }
    b[c]=d / a[c][c];
  }
}

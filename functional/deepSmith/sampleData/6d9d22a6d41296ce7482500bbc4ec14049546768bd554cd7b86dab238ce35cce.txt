private static boolean fn_A(int[] a,int b){
  if (a.length == 0) {
    return false;
  }
  final int c=a.length - 1;
  int d=c;
  while (d >= 0) {
    a[d]++;
    if (a[d] >= (b - (c - d))) {
      d--;
    }
 else {
      break;
    }
  }
  if (d < 0) {
    return false;
  }
  for (; d < c; d++) {
    a[d + 1]=a[d] + 1;
  }
  return true;
}

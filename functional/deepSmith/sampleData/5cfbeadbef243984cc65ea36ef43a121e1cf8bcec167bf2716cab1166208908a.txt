public static float fn_A(float a,float b,float c,float d,float e,float f){
  c-=a;
  d-=b;
  e-=a;
  f-=b;
  float g=e * c + f * d;
  float h;
  if (g <= 0.0) {
    h=0.0f;
  }
 else {
    e=c - e;
    f=d - f;
    g=e * c + f * d;
    if (g <= 0.0) {
      h=0.0f;
    }
 else {
      h=g * g / (c * c + d * d);
    }
  }
  float i=e * e + f * f - h;
  if (i < 0) {
    i=0;
  }
  return i;
}

public final static float fn_A(float[] a,int b,int c){
  int d, e;
  int f=0;
  int g=b - 1;
  float h=a[c];
  float i;
  while (f < g) {
    d=f;
    e=g;
    do {
      while (a[d] < h)       d++;
      while (h < a[e])       e--;
      i=a[e];
      a[e]=a[d];
      a[d]=i;
      d++;
      e--;
    }
 while ((e >= c) && (d <= c));
    if (e < c)     f=d;
    if (c < d)     g=e;
    h=a[c];
  }
  return h;
}

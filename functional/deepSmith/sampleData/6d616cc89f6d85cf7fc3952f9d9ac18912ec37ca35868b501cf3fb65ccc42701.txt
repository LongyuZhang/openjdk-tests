public static int[] fn_A(final int a,final boolean b){
  if (a <= 0) {
    throw new IllegalArgumentException("Specified max for indices must be more than 0, got: " + a);
  }
  int c;
  if (b) {
    c=a + 1;
  }
 else {
    c=a;
  }
  int[] d=new int[c];
  for (int e=0; e < c; e++) {
    d[e]=e;
  }
  return d;
}

  c|=(long)(a[++b] & 0xff) << 8;
  c|=(long)(a[++b] & 0xff) << 16;
  c|=(long)(a[++b] & 0xff) << 24;
  c|=(long)(a[++b] & 0xff) << 32;
  c|=(long)(a[++b] & 0xff) << 40;
  c|=(long)(a[++b] & 0xff) << 48;
  c|=(long)(a[++b] & 0xff) << 56;
  return c;
}

public static boolean fn_A(byte[] a,byte[] b){
  if (a == null || b == null) {
    return false;
  }
  if (a == b) {
    return true;
  }
  int c=(a.length < b.length) ? a.length : b.length;
  int d=a.length ^ b.length;
  for (int e=0; e != c; e++) {
    d|=(a[e] ^ b[e]);
  }
  for (int e=c; e < b.length; e++) {
    d|=(b[e] ^ ~b[e]);
  }
  return d == 0;
}

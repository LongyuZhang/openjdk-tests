  a=((a & 0xCCCCCCCCCCCCCCCCL) >> 2) + (a & 0x3333333333333333L);
  a=((a & 0x3030303030303030L) >> 4) + (a & 0x0303030303030303L);
  a=((a & 0x0700070007000700L) >> 8) + (a & 0x0007000700070007L);
  a=((a & 0x000F0000000F0000L) >> 16) + (a & 0x0000000F0000000FL);
  a=((a & 0x0000001F00000000L) >> 32) + (a & 0x000000000000001FL);
  return (int)a;
}

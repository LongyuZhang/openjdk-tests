public static void main(String a[]){
  int[] b={8,9,3,5,1};
  int c=0;
  int d=0;
  System.out.println("Vetor desordenado: ");
  for (d=0; d < 5; d++) {
    System.out.println(" " + b[d]);
  }
  System.out.println(" ");
  for (d=0; d < 5; d++) {
    for (int e=0; e < 4; e++) {
      if (b[e] > b[e + 1]) {
        c=b[e];
        b[e]=b[e + 1];
        b[e + 1]=c;
      }
    }
  }
  System.out.println("Vetor organizado:");
  for (d=0; d < 5; d++) {
    System.out.println(" " + b[d]);
  }
}

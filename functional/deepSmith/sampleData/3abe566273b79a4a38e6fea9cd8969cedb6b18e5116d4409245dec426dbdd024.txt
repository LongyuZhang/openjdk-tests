public static boolean fn_A(float a,float b,float c,float d,float e,float f,float g,float h,float i,float j){
  float k=g - a;
  float l=h - b;
  float m=i - c;
  float n=k * d + l * e + m * f;
  float o=k * k + l * l + m * m - n * n;
  if (o > j)   return false;
  float p=(float)Math.sqrt(j - o);
  float q=n - p;
  float r=n + p;
  return q < r && r >= 0.0f;
}

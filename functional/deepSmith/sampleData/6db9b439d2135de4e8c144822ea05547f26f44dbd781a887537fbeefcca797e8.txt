public static float fn_A(final float a){
  if (a < Float.POSITIVE_INFINITY) {
    final int b=Float.floatToRawIntBits(a + 0.0f);
    return Float.intBitsToFloat(b + (b >= 0 ? 1 : -1));
  }
 else   if (a == Float.POSITIVE_INFINITY) {
    return Float.POSITIVE_INFINITY;
  }
 else {
    return a;
  }
}

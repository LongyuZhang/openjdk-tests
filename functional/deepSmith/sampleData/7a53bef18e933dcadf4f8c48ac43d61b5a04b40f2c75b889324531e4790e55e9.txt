public static void fn_A(final Object... a) throws IllegalArgumentException {
  if (a.length % 2 != 0)   throw new IllegalArgumentException("The provided arguments must have a size that is a factor of 2");
  for (int b=0; b < a.length; b=b + 2) {
    if (!(a[b] instanceof String))     throw new IllegalArgumentException("The provided key/value array must have a String key on even array indices");
  }
}

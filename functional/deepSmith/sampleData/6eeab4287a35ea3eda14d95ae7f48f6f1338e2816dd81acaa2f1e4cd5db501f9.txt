public static boolean fn_A(int a,int b,int[][] c,int[][] d,int e){
  int f=a - d.length / 2;
  int g=b - d[0].length / 2;
  int h=g;
  for (int i=0; i < d.length; i++) {
    g=h;
    for (int j=0; j < d[0].length; j++) {
      if (f >= 0 && f < c.length && g >= 0 && g < c[0].length) {
        if (d[i][j] == e && c[f][g] == e) {
          return true;
        }
      }
      g++;
    }
    f++;
  }
  return false;
}

public static byte[] fn_A(int a,boolean b){
  byte[] c=new byte[4];
  if (b) {
    c[0]=(byte)(a >>> 24);
    c[1]=(byte)(a >>> 16 & 0xff);
    c[2]=(byte)(a >>> 8 & 0xff);
    c[3]=(byte)(a & 0xff);
  }
 else {
    c[0]=(byte)(a & 0xff);
    c[1]=(byte)(a >>> 8 & 0xff);
    c[2]=(byte)(a >>> 16 & 0xff);
    c[3]=(byte)(a >>> 24);
  }
  return c;
}

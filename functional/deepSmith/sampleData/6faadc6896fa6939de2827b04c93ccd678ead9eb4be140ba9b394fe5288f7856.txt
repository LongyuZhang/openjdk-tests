public static long fn_A(int[] a,int[] b,int[] c){
  int d=0, e=0;
  long f=0L;
  while (d < b.length || e < c.length) {
    if (d == b.length) {
      a[d + e]=c[e];
      e++;
    }
 else     if (e == c.length) {
      a[d + e]=b[d];
      d++;
    }
 else     if (b[d] <= c[e]) {
      a[d + e]=b[d];
      d++;
    }
 else {
      a[d + e]=c[e];
      f=f + b.length - d;
      e++;
    }
  }
  return f;
}

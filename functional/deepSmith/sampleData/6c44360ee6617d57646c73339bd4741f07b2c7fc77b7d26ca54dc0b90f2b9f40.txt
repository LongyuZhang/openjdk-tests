private static byte[] fn_A(byte[] a,long b){
  long c=b;
  int d=1;
  if (b < 0) {
    c=-b;
    d=-1;
  }
  for (int e=0; e < a.length; e++) {
    int f=((int)c % 256) * d;
    c=(c >> 8);
    int g=((~a[a.length - e - 1]) & 0x0ff) + 1;
    int h=f - g;
    if (h >= 0) {
      c+=d;
    }
 else     if (h < -256) {
      c-=d;
      h%=256;
    }
    a[a.length - e - 1]=(byte)h;
    if (c == 0)     return a;
  }
  return a;
}

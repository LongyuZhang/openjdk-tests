  long d=a[b + 1] & 255;
  long e=a[b + 2] & 255;
  long f=a[b + 3] & 255;
  long g=a[b + 4] & 255;
  long h=a[b + 5] & 255;
  long i=a[b + 6] & 255;
  long j=a[b + 7] & 255;

}

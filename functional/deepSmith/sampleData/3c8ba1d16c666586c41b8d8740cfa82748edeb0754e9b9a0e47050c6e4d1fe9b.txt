private static int fn_A(byte[] a,byte[] b){
  for (int c=0; c < a.length; c++) {
    if (c == b.length) {
      return 1;
    }
    if (a[c] == b[c]) {
      continue;
    }
    if (a[c] > b[c]) {
      return 1;
    }
 else {
      return -1;
    }
  }
  if (b.length > a.length) {
    return -1;
  }
  return 0;
}

public static boolean fn_A(char a){
switch (a) {
case '~':
case '!':
case '@':
case '#':
case '$':
case '%':
case '^':
case '&':
case '*':
case '(':
case ')':
case '_':
case '+':
case '-':
case '=':
case '|':
case '\\':
case '<':
case '>':
case '?':
case '/':
case ',':
case '.':
case '`':
case '\'':
case ';':
case ':':
case '[':
case ']':
case '{':
case '}':
case '"':
    return true;
default :
  return false;
}
}

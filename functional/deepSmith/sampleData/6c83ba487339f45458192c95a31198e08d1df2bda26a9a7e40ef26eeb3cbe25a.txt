  b[c + 3]=(byte)((a >>> 32) & 0xFF);
  b[c + 4]=(byte)((a >>> 24) & 0xFF);
  b[c + 5]=(byte)((a >>> 16) & 0xFF);
  b[c + 6]=(byte)((a >>> 8) & 0xFF);
  b[c + 7]=(byte)(a & 0xFF);
}

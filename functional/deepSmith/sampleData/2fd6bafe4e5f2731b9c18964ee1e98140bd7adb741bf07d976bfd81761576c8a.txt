private static int fn_A(byte[] a,int b,int c){
  if (b >= c) {
    return 0;
  }
  while (b < c && a[b] != '\n') {
    if (a[b] >= '0' && a[b] <= '9') {
      int d=b++;
      while (b < c && a[b] >= '0' && a[b] <= '9') {
        ++b;
      }
      return Integer.parseInt(new String(a,d,b - d),10);
    }
    ++b;
  }
  return 0;
}

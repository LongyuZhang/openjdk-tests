private static int fn_A(int[] a){
  int b=Integer.MAX_VALUE;
  int c=Integer.MIN_VALUE;
  if (a != null) {
    for (int d=0, e=a.length; d < e; d+=2) {
      int f=a[d + 0];
      int g=a[d + 1];
      if (f < b) {
        b=f;
      }
      if (g > c) {
        c=g;
      }
    }
    if (c < 0) {
      c=0;
    }
    if (b > c) {
      b=c;
    }
  }
  return b;
}

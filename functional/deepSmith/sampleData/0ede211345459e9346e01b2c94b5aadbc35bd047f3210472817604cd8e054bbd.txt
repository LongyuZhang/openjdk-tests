public static int fn_A(int[] a){
  int b=0;
  int c=b + 1;
  int d=a.length - 1;
  int e=a[b];
  while (d > c) {
    while (c <= d && a[c] <= e)     c++;
    while (c <= d && a[d] > e)     d--;
    if (d > c) {
      int f=a[d];
      a[d]=a[c];
      a[c]=f;
    }
  }
  while (d >= c && a[d] >= e)   d--;
  if (d > b) {
    int f=a[d];
    a[d]=a[b];
    a[b]=f;
    return d;
  }
 else {
    return b;
  }
}

static int fn_A(int[] a){
  int b=1000;
  for (int c=0; c < a.length; c++) {
    int d=c;
    for (int e=c + 1; e < a.length; e++) {
      if (a[e] == a[d]) {
        d=e;
        if (Math.abs(c - e) < b) {
          b=Math.abs(c - e);
        }
        if (Math.abs(c - e) == 1) {
          return 1;
        }
      }
    }
  }
  if (b == 1000) {
    return -1;
  }
  return b;
}

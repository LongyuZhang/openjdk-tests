private static boolean fn_A(float[] a,float[] b,float[] c,float[] d){
  for (int e=0; e < 4; ++e) {
    int f=e + 1 & 3;
    float g=a[e] - a[f];
    float h=b[e] - b[f];
    float i=-h;
    float j=g;
    if (i != 0 || g != 0) {
      for (int k=0; k < 4; ++k) {
        float l=i * (c[k] - a[f]) + j * (d[k] - b[f]);
        if (l < 0) {
          continue;
        }
      }
      return true;
    }
  }
  return false;
}

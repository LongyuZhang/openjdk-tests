public static int fn_A(final int a){
  if (a < 10)   return 1;
  if (a < 100)   return 2;
  if (a < 1000)   return 3;
  if (a < 10000)   return 4;
  if (a < 100000)   return 5;
  if (a < 1000000)   return 6;
  if (a < 10000000)   return 7;
  if (a < 100000000)   return 8;
  if (a < 1000000000)   return 9;
  return 10;
}

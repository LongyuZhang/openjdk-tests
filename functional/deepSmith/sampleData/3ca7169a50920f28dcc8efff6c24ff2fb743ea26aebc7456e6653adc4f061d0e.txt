public static long fn_A(final byte[] a,final int b){
  long c=((long)a[b] & 0xFF);
  c|=((long)a[b + 1] & 0xFF) << 8;
  c|=((long)a[b + 2] & 0xFF) << 16;
  c|=((long)a[b + 3] & 0xFF) << 24;
  c|=((long)a[b + 4] & 0xFF) << 32;
  c|=((long)a[b + 5] & 0xFF) << 40;
  c|=((long)a[b + 6] & 0xFF) << 48;

}

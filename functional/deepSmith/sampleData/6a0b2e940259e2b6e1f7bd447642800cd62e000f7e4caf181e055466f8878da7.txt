static void fn_A(long[] a,int b,byte[] c,int d,int e){
  e+=d;
  while (d < e) {
    long f=a[b++];
    c[d++]=(byte)(f >> 56);
    c[d++]=(byte)(f >> 48);
    c[d++]=(byte)(f >> 40);
    c[d++]=(byte)(f >> 32);
    c[d++]=(byte)(f >> 24);
    c[d++]=(byte)(f >> 16);
    c[d++]=(byte)(f >> 8);
    c[d++]=(byte)(f);
  }
}

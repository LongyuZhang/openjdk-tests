public static void fn_A(char[] a,int b){
  int c=0, d, e=0;
  for (e=0; e < b; e++) {
    if (a[e] == ' ') {
      c++;
    }
  }
  d=b + c * 2;
  if (b < a.length)   a[b]='\0';
  for (e=b - 1; e >= 0; e--) {
    if (a[e] == ' ') {
      a[d - 1]='0';
      a[d - 2]='2';
      a[d - 3]='%';
      d=d - 3;
    }
 else {
      a[d - 1]=a[e];
      d--;
    }
  }
}

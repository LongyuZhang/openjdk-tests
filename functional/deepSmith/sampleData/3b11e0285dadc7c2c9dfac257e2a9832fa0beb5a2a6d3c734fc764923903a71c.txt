public static Date fn_A(byte[] a){
  assert(a != null && (a.length == 5));
  if (a == null)   throw new IllegalArgumentException("pb");
  if (a.length != 5)   throw new IllegalArgumentException();
  int b=a[0], c=a[1], d=a[2], e=a[3], f=a[4];
  int g=(b << 6) | (c >> 2);
  int h=((c & 0x00000003) << 2) | (d >> 6);
  int i=(d >> 1) & 0x0000001F;
  int j=((d & 0x00000001) << 4) | (e >> 4);
  int k=((e & 0x0000000F) << 2) | (f >> 6);
  int l=f & 0x0000003F;
  return new Date(g,h,i,j,k,l);
}

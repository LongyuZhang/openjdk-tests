private static void fn_A(long[][] a,int b,int c){
  boolean d=false;
  for (int e=0; e < b; e++) {
    for (int f=0; f < c + 1; f++) {
      if (a[e][f] == 0) {
        System.err.println("ERROR: Count of (adjusted) counts for order " + (e + 1) + ", count category "+ (f + 1)+ " is zero.");
        d=true;
      }
    }
  }
  if (d) {
    System.err.println("FATAL: Count of counts are not valid. Did you give me artificial or very small data?");
    System.exit(1);
  }
}

public static float[] fn_A(int a,boolean b){
  float[] c=new float[a];
  int d;
  if (b) {
    d=a + 1;
    for (int e=1; e < a; e++) {
      c[e]=(float)(0.5 * (1.0 - Math.cos((2.0 * Math.PI * (e)) / (d - 1))));
    }
  }
 else {
    d=a + 2;
    for (int e=0; e < a; e++) {
      c[e]=(float)(0.5 * (1.0 - Math.cos((2.0 * Math.PI * (e + 1)) / (d - 1))));
    }
  }
  return c;
}

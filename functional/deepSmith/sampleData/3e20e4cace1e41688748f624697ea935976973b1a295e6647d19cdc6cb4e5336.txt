private static boolean fn_A(int a,int b,int c,int[][] d){
  for (int e=0; e < 9; ++e)   if (d[a][e] == c)   return false;
  for (int f=0; f < 9; ++f)   if (d[f][b] == c)   return false;
  int g=(a / 3) * 3;
  int h=(b / 3) * 3;
  for (int f=0; f < 3; ++f) {
    for (int e=0; e < 3; ++e) {
      if (d[g + f][h + e] == c)       return false;
    }
  }
  return true;
}

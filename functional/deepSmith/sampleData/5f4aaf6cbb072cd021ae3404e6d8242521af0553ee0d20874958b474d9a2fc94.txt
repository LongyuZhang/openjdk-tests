private static int fn_A(byte[] a,int b){
  int c=0;
  c|=(a[0] ^ 2);
  int d=a.length - (b + 1);
  for (int e=1; e < d; e++) {
    int f=a[e];
    f|=f >> 1;
    f|=f >> 2;
    f|=f >> 4;
    c|=(f & 1) - 1;
  }
  c|=a[a.length - (b + 1)];
  c|=c >> 1;
  c|=c >> 2;
  c|=c >> 4;
  return ~((c & 1) - 1);
}

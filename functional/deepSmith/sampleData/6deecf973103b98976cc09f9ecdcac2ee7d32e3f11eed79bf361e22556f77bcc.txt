public static double[][] fn_A(double a,int b,float c,double d,int e) throws Exception {
  if (d <= 0.0)   throw new Exception("Invalid maximum allowable fraction of sampling rate of " + d + " specified.\n"+ "This value must be above 0.");
  if (a <= 0.0)   throw new Exception("Invalid fundamental frequence of " + a + " Hz specified.\n"+ "Frequency must be above 0 Hz.");
  if (a >= (d * c))   throw new Exception("Invalid fundamental frequency of " + a + " Hz specified.\n"+ "Frequency must be below "+ (d * c)+ " Hz\n"+ "under current settings. This is done in order to avoid aliasing at this\n"+ "sampling rate of "+ c+ " Hz for this type of synthesis.");
  if (b < 1)   throw new Exception("There must be 1 or more channels. You specified " + b + ".");
  if (c <= 0.0F)   throw new Exception("Invalid sampling rate of " + c + " Hz specified.\n"+ "Must be greater than 0.");
  if (e <= 0)   throw new Exception("Invalid total number of samples per channel of " + e + " specified.\n"+ "Must be greater than 0.");
  double[][] f=new double[b][e];
  double g=a;
  double h=g / 10.0;
  for (int i=0; i < e; i++) {
    double j=i / c;
    double k=(double)i / (double)e;
    double l=h + (g - h) * k;
    double m=(Math.sin(2 * Math.PI * l* j));
    for (int n=0; n < f.length; n++)     f[n][i]=m;
  }
  return f;
}

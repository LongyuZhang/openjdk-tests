static void fn_A(int a[]){
  int b=a.length - 2;
  int c=a[0];
  for (int d=1; d < a.length; d++)   c^=a[d];
  for (int d=1; d <= b; d++)   c^=d;
  int e=(c & ~(c - 1));
  int f=0, g=0;
  for (int d=0; d < a.length; d++) {
    int h=a[d] & e;
    if (h != 0)     f=f ^ a[d];
 else     g=g ^ a[d];
  }
  for (int d=1; d <= b; d++) {
    int h=d & e;
    if (h != 0)     f=f ^ d;
 else     g=g ^ d;
  }
  System.out.println(f + " " + g);
}

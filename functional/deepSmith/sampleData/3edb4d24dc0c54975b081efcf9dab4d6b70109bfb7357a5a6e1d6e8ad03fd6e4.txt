static int[] fn_A(int[] a,int[] b){
  int[] c=new int[a.length + b.length];
  int d=0;
  int e=0;
  int f=0;
  while (e < a.length || f < b.length) {
    if (e < a.length && f < b.length) {
      int g=a[e];
      int h=b[f];
      if (g < h) {
        c[d]=g;
        e++;
      }
 else {
        c[d]=h;
        f++;
      }
    }
 else     if (e < a.length) {
      c[d]=a[e];
      e++;
    }
 else {
      c[d]=b[f];
      f++;
    }
    d++;
  }
  return c;
}

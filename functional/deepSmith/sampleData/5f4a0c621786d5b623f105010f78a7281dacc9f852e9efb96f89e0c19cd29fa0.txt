public static int fn_A(char[] a,char[] b){
  int c[]=new int[256];
  for (int d=0; d < 256; d++) {
    c[d]=a.length;
  }
  for (int d=0; d < a.length - 1; d++) {
    c[a[d]]=a.length - 1 - d;
  }
  int e=0, f=0;
  while ((e + a.length) <= b.length) {
    f=a.length - 1;
    while (b[e + f] == a[f]) {
      f-=1;
      if (f < 0)       return e;
    }
    e=e + c[b[e + a.length - 1]];
  }
  return -1;
}

public static double[] fn_A(double[] a,int b){
  double[] c=new double[a.length];
  double d=0.0;
  double e=0.0;
  int f=0;
  int g=0;
  for (int h=0; h < b; h++) {
    d=0.0;
    f=h * b;
    for (int i=0; i < h; i++) {
      e=0.0;
      g=i * b;
      for (int j=0; j < i; j++) {
        e+=c[f + j] * c[g + j];
      }
      c[f + i]=(a[f + i] - e) / c[g + i];
      d+=c[f + i] * c[f + i];
    }
    c[f + h]=Math.sqrt(a[f + h] - d);
  }
  return c;
}

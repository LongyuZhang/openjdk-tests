private static int fn_A(final double[][] a,final boolean[] b,final boolean[] c){
  int d=0;
  for (int e=0; e < a.length; ++e) {
    if (b[e]) {
      continue;
    }
    for (int f=0; f < a[e].length; ++f) {
      if (c[f]) {
        continue;
      }
      if (Double.valueOf(a[e][f]).compareTo(0.) == 0) {
        ++d;
      }
    }
  }
  return d;
}

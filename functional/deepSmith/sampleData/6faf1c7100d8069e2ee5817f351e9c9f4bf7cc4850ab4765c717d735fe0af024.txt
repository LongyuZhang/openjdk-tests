public static String fn_A(long a){
  if (a < 5 * 1000) {
    return a + "ms";
  }
 else   if (a < 3 * 60 * 1000) {
    return (a / 1000) + "s";
  }
 else   if (a < 120 * 60 * 1000) {
    return (a / (60 * 1000)) + "m";
  }
 else   if (a < 3 * 24 * 60* 60* 1000) {
    return (a / (60 * 60 * 1000)) + "h";
  }
 else   if (a > 1000l * 24l * 60l* 60l* 1000l) {
    return "n/a";
  }
 else {
    return (a / (24 * 60 * 60* 1000)) + "d";
  }
}

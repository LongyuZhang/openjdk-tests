public static double fn_A(int a){
  if (a >= 2500) {
    return 0.0075;
  }
 else   if (a >= 2400) {
    return 0.009;
  }
 else   if (a >= 2351) {
    return 0.00975;
  }
 else   if (a >= 2251) {
    return 0.0105;
  }
 else   if (a >= 2201) {
    return 0.0120;
  }
 else   if (a >= 2151) {
    return 0.0135;
  }
 else {
    return 0.015;
  }
}

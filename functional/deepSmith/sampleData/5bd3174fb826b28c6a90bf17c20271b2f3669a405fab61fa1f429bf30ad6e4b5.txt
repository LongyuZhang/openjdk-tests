private static void fn_A(final int[] a,final long[] b,final long[] c,final int d,final int e){
  for (int f=d; ++f < e; ) {
    int g=a[f];
    int h=f;
    for (int i=a[h - 1]; ((b[g]) < (b[i])) || ((b[g]) == (b[i])) && ((c[g]) < (c[i])); i=a[--h - 1]) {
      a[h]=i;
      if (d == h - 1) {
        --h;
        break;
      }
    }
    a[h]=g;
  }
}

public static boolean fn_A(double a,double b,double c,double d,double e,double f,double g,double h){
  double i=g;
  double j=h;
  if (i <= 0 || j <= 0 || c <= 0 || d <= 0) {
    return false;
  }
  double k=e;
  double l=f;
  i+=k;
  j+=l;
  c+=a;
  d+=b;
  return ((i < k || i > a) && (j < l || j > b) && (c < a || c > k)&& (d < b || d > l));
}

static Integer fn_A(Object a){
  if (a instanceof Integer) {
    return (Integer)a;
  }
 else   if (a instanceof Number) {
    return ((Number)a).intValue();
  }
 else   if (a instanceof String) {
    try {
      return (int)Double.parseDouble((String)a);
    }
 catch (    NumberFormatException b) {
    }
  }
  return null;
}

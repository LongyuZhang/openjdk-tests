private static final boolean fn_A(double a,double b){
  if (a == b)   return true;
  final double c=1.0e-6;
  final double d=1.0e-4;
  double e=Math.abs(a - b);
  double f=Math.abs(a);
  double g=Math.abs(b);
  double h=(f >= g) ? f : g;
  if (e < c)   return true;
  if ((e / h) < d)   return true;
  return false;
}

public static void main(String[] a){
  int b=13;
  int c=9;
  int d=37;
  int e=(b * 60 + c) * 60 + d;
  int f=24 * 60 * 60;
  int g=f - e;
  int h=e * 100 / f;
  System.out.println("Seconds since midnight = " + e);
  System.out.println("Seconds until midnight = " + g);
  System.out.println("Percentage of day elapsed = " + h);
}

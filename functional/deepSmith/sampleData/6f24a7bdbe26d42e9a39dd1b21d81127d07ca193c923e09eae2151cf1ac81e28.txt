private static float[] fn_A(int a){
  final float[] b=new float[a];
  final int c=a / 8;
  final int d=a / 2;
  final double e=12.566370614359172D / (a - 1);
  int f=1;
  int g=a - 2;
  while (f <= c) {
    final double h=Math.sin(e * f);
    b[f++]=b[g--]=(float)(h * h);
  }
  while (f <= d) {
    b[f++]=b[g--]=1f;
  }
  return b;
}

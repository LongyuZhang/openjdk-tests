public static long fn_A(byte[] a,boolean b,boolean c){
  long d;
  if (b) {
    if (c)     d=a[0];
 else     d=(a[0] >= 0 ? a[0] : a[0] + 256);
    for (int e=1; e < a.length; e++)     d=256 * d + (a[e] >= 0 ? a[e] : a[e] + 256);
  }
 else {
    if (c)     d=a[a.length - 1];
 else     d=(a[a.length - 1] >= 0 ? a[a.length - 1] : a[a.length - 1] - Byte.MIN_VALUE);
    for (int e=a.length - 2; e >= 0; e--)     d=256 * d + (a[e] >= 0 ? a[e] : a[e] - Byte.MIN_VALUE);
  }
  return d;
}

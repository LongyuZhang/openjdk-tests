private static final void fn_A(int[] a,int[] b,int c,int d,int e,int f,int g,int h){
  for (int i=-f; i < 0; i++) {
    int j=d + e;
    while (d < j) {
      int k=b[c] >>> 24;
      int l=256 - k;
      int m=b[c++];
      int n=a[d];
      a[d++]=((((m & 0xff00ff) * k + (n & 0xff00ff) * l) & ~0xff00ff) + (((m & 0xff00) * k + (n & 0xff00) * l) & 0xff0000)) >> 8;
    }
    d+=g;
    c+=h;
  }
}

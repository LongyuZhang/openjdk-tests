static int[][] fn_A(int a){
  int[][] b=new int[a][a];
  for (int c=0; c < a / 2 + a % 2; c++) {
    for (int d=0; d < a; d++) {
      b[c][d]=(c == 0 || d == 0 || c == a - 1 || d == a - 1) ? 1 : c < a / 2 + a % 2 ? (c + 1) : 0;
      b[a - c - 1][d]=b[c][d];
    }
  }
  if (a % 2 == 1 && a > 1) {
    b[a / 2][a / 2]=a - 1;
  }
  return b;
}

public static int fn_A(byte[] a,int b,int c,byte d){
  if (b > c)   throw new IllegalArgumentException("The start index is higher than " + "the finish index.");
  if (b < 0 || c > a.length)   throw new ArrayIndexOutOfBoundsException("One of the indices is out " + "of bounds.");
  int e=0;
  while (b <= c) {
    e=(b + c) >>> 1;
    final byte f=a[e];
    if (f == d)     return e;
 else     if (f > d)     c=e - 1;
 else     b=++e;
  }
  return -e - 1;
}

public static short fn_A(byte[] a){
  if (a == null) {
    throw new IllegalArgumentException("input: parameter may not be null.");
  }
  if (a.length != 2) {
    throw new IllegalArgumentException("input: Byte array length must be 2.");
  }
  short b=0;
  b=(short)(((a[0] & 0xff) << 8) | (a[1] & 0xff));
  return b;
}

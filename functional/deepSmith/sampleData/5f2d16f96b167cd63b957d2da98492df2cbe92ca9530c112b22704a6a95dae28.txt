public static double[] fn_A(final int a,final double b){
  int c=3;
  final double[] d;
  if (b <= 0) {
    d=new double[3];
    d[1]=1;
  }
 else {
    c=Math.max(3,(2 * (int)(a + 0.5) + 1));
    final double e=2 * b * b;
    final double f=Math.sqrt(2 * Math.PI) * b;
    d=new double[c];
    for (int g=c / 2; g >= 0; --g) {
      final double h=Math.exp(-(g * g) / e) / f;
      d[c / 2 - g]=h;
      d[c / 2 + g]=h;
    }
  }
  return d;
}

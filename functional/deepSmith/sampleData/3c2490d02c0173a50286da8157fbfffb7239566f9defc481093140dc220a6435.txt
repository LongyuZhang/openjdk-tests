public static int fn_A(char[] a,char[] b){
  if (a == null || b == null) {
    return -1;
  }
  int c=a.length;
  int d=b.length;
  while (c > 0 && d > 0 && a[c - 1] == b[d - 1]) {
    c--;
    d--;
  }
  int e=d;
  for (int f=0; f < c; f++) {
    if (f + e > c) {
      e=c - f;
    }
    int g;
    for (g=0; g < e; g++) {
      if (a[f + g] != b[g]) {
        break;
      }
    }
    if (g == e) {
      return f;
    }
  }
  return -1;
}

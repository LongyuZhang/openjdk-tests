public static void main(String[] a) throws InterruptedException, IOException {
  int b=0;
  int c=0;
  while (true) {
    System.out.print((char)('a' + b));
    System.out.print(" ");
    Thread.sleep(50);
    b++;
    if (b == 26) {
      System.out.print("\n");
      b=0;
    }
    System.out.print(c++ + " ");
    if (c % 20 == 0)     System.out.print("\n");
    if (System.in.available() > 0) {
      System.exit(0);
    }
  }
}

public static void main(String[] a){
  int b=60;
  int c=13;
  int d=0;
  d=b & c;
  System.out.println("a & b = " + d);
  d=b | c;
  System.out.println("a | b = " + d);
  d=b ^ c;
  System.out.println("a ^ b = " + d);
  d=~b;
  System.out.println("~a = " + d);
  d=b << 2;
  System.out.println("a << 2 = " + d);
  d=b >> 2;
  System.out.println("a >> 2  = " + d);
  d=b >>> 2;
  System.out.println("a >>> 2 = " + d);
}

private static void fn_A(int[] a,int[] b,int c,int d,int e){
  int f=c;
  int g=d + 1;
  int h=0;
  while (f <= d && g <= e) {
    if (a[f] < a[g]) {
      b[h++]=a[f++];
    }
 else {
      b[h++]=a[g++];
    }
  }
  while (f <= d) {
    b[h++]=a[f++];
  }
  while (g <= e) {
    b[h++]=a[g++];
  }
  h=0;
  while (c <= e) {
    a[c++]=b[h++];
  }
}

private static boolean fn_A(Class a){
  if (a == java.lang.Boolean.TYPE)   return true;
  if (a == java.lang.Character.TYPE)   return true;
  if (a == java.lang.Byte.TYPE)   return true;
  if (a == java.lang.Short.TYPE)   return true;
  if (a == java.lang.Integer.TYPE)   return true;
  if (a == java.lang.Long.TYPE)   return true;
  if (a == java.lang.Float.TYPE)   return true;
  if (a == java.lang.Double.TYPE)   return true;
  if (a == java.lang.Void.TYPE)   return true;
  return false;
}

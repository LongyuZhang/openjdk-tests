private static int fn_A(int[] a){
  if (a == null)   throw new NullPointerException("null array");
  if (a.length == 0)   throw new IllegalArgumentException("Zero-length array");
  int b=0;
  for (int c=0; c < a.length; c++) {
    if (a[c] <= 0) {
      throw new IllegalArgumentException("All array values must be > 0");
    }
    b+=a[c];
  }
  return b;
}

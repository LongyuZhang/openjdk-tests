public static String fn_A(long a){
  final long[] b=new long[]{1L << 30,1L << 20,1L << 10,1};
  final String[] c=new String[]{"GB","MB","KB","B","B"};
  if (a <= 0) {
    return String.format("0 B");
  }
  int d=0;
  for (; d < b.length; ++d) {
    if (a >= b[d])     break;
  }
  return String.format("%.2f %s",(a * 1.0f) / b[d],c[d]);
}

public static void fn_A(int[][] a){
  for (int b=0; b < a.length; b++) {
    for (int c=0; c < b; c++) {
      int d=a[b][c];
      a[b][c]=a[c][b];
      a[c][b]=d;
    }
  }
  for (int b=0; b < a.length; b++) {
    for (int c=0; c < a[0].length / 2; c++) {
      int d=a[b][c];
      a[b][c]=a[b][a[0].length - 1 - c];
      a[b][a[0].length - 1 - c]=d;
    }
  }
}

private static byte fn_A(byte a){
  if ((a >= '0') && (a <= '9'))   return (byte)(a - '0');
  if ((a >= 'a') && (a <= 'f'))   return (byte)(a - 'a' + 10);
  if ((a >= 'A') && (a <= 'F'))   return (byte)(a - 'A' + 10);
  throw new IllegalArgumentException(((char)a) + " is not a hexadecimal digit");
}

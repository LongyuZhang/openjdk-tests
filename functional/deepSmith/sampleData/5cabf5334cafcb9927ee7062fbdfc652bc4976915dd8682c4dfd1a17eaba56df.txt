private static double[] fn_A(double[] a){
  double b=3.5;
  double[] c=new double[a.length];
  double d=2 * Math.PI / (a.length - 1);
  double e=0;
  double f=0;
  for (int g=0; g < c.length; g++) {
    c[g]=(a[g] * (0.42 - 0.5 * Math.cos(d * g) + 0.08 * Math.cos(d * g))) * 3.5;
    if (c[g] < f) {
      f=c[g];
    }
    if (c[g] > e) {
      e=c[g];
    }
  }
  return c;
}

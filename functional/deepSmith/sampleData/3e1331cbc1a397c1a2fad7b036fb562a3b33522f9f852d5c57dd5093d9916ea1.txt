public static void fn_A(int a[][]){
  for (int b=0; b < a.length - 1; b++) {
    int c=b;
    int[] d=a[b];
    for (int e=b + 1; e < a.length; e++) {
      if (d[0] > a[e][0] || d[0] == a[e][0] && d[1] > a[e][1]) {
        d=a[e];
        c=e;
      }
    }
    if (c != b) {
      a[c]=a[b];
      a[b]=d;
    }
  }
}

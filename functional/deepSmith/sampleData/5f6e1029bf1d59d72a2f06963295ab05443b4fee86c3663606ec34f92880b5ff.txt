public static boolean fn_A(final char a){
switch (a) {
case '"':
case '?':
case '%':
case '&':
case '\'':
case '(':
case ')':
case '|':
case '*':
case '+':
case ',':
case '-':
case '.':
case '/':
case ':':
case ';':
case '<':
case '=':
case '>':
case '!':
case '~':
case '`':
case '[':
case ']':
case '#':
    return true;
default :
  return false;
}
}

public static String fn_A(int a){
  int b=0;
  int c=0;
  String d="";
  a/=1000;
  if (a < 60) {
    b=0;
    c=a;
  }
 else {
    b=a / 60;
    c=a % 60;
  }
  if (b < 10) {
    d+="0" + b + ":";
  }
 else {
    d+=b + ":";
  }
  if (c < 10) {
    d+="0" + c;
  }
 else {
    d+=c;
  }
  return d;
}

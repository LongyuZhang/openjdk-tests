public static byte[] fn_A(final int a){
  if ((a & 0xC00000) == 0xC00000) {
    return new byte[]{(byte)((a >> 16) & 0xFF),(byte)((a >> 8) & 0xFF),(byte)(a & 0xFF)};
  }
 else   if ((a & 0xFF8000) == 0x8000) {
    return new byte[]{(byte)((a >> 8) & 0xFF),(byte)(a & 0xFF)};
  }
 else {
    return new byte[]{(byte)a};
  }
}

public static int fn_A(int[] a,int b,int c){
  int d=0;
  int e=b - 2;
  int f=0;
  while (d <= e) {
    f=(d + e) / 2;
    if ((f % 2) != 0)     f--;
    int g=a[f];
    if (c < g) {
      e=f - 2;
    }
 else     if (c > g) {
      d=f + 2;
    }
 else {
      return -1;
    }
  }
  if (c < a[f])   return f;
  return f + 2;
}

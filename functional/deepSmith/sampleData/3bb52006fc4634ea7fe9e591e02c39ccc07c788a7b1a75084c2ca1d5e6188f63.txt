public static int fn_A(byte[] a,int b,boolean c){
  int d=0;
  int e, f, g, h;
  e=a[0 + b] < 0 ? a[0 + b] + 256 : a[0 + b];
  f=a[1 + b] < 0 ? a[1 + b] + 256 : a[1 + b];
  g=a[2 + b] < 0 ? a[2 + b] + 256 : a[2 + b];
  h=a[3 + b] < 0 ? a[3 + b] + 256 : a[3 + b];
  if (!c)   d=(h << 24) + (g << 16) + (f << 8)+ e;
 else   d=(e << 24) + (f << 16) + (g << 8)+ h;
  return d;
}

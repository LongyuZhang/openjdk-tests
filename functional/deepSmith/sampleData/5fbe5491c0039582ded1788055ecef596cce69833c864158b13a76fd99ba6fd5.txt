public static void main(String[] a){
  int b, c;
  b=17;
  System.out.println("Before incrementing, myNumber is " + b);
  c=++b;
  System.out.println("After prefix increment, myNumber is " + b);
  System.out.println("  and answer is " + c);
  b=17;
  System.out.println("Before incrementing, myNumber is " + b);
  c=b++;
  System.out.println("After postfix increment, myNumber is " + b);
  System.out.println("  and answer is " + c);
}

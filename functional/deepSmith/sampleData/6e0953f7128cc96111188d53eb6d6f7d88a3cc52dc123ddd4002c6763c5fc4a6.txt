public static void main(String... a){
  int b=new Scanner(System.in).nextInt();
  int c=0;
  for (int d=1; d <= b; d++) {
    for (int e=1; e <= b; e++) {
      int f=d ^ e;
      if (d == 0 || e == 0 || f == 0 || d > b || e > b || f > b)       continue;
      if (d <= f && e <= f && f < d + e)       c++;
 else       if (e <= d && f <= d && d < e + f)       c++;
 else       if (f <= e && d <= e && e < d + f)       c++;
    }
  }
  System.out.print(c / 6);
}

static double fn_A(double a){
  if (a < 1.0 / 2.75) {
    return 7.5625 * a * a;
  }
 else   if (a < 2 / 2.75) {
    a-=1.5 / 2.75;
    return 7.5625 * a * a + 0.75;
  }
 else   if (a < 2.5 / 2.75) {
    a-=2.25 / 2.75;
    return 7.5625 * a * a + 0.9375;
  }
  a-=2.625 / 2.75;
  return 7.5625 * a * a + 0.984375;
}

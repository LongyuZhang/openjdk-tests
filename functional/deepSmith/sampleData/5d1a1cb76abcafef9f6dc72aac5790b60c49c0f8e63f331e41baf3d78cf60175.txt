private static final int fn_A(int[][] a,int b,int c){
  int d=b >> 7;
  int e=c >> 7;
  if (d < 0 || e < 0 || d >= a.length || e >= a[0].length)   return 0;
  int f=b & 0x7f;
  int g=c & 0x7f;
  int h=((a[d][e] * (128 - f) + a[d + 1][e] * f) >> 7);
  int i=((a[d][e + 1] * (128 - f) + a[d + 1][e + 1] * f) >> 7);
  return h * (128 - g) + i * g >> 7;
}

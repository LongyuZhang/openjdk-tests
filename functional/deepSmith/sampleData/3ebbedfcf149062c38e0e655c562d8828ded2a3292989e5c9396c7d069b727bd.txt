public static double[] fn_A(int a,double b,double c,double d,double e){
  double[] f=new double[a];
  double g, h, i, j, k;
  for (int l=1; l < a; l++) {
    g=(l - a / 2) * 2.0 * Math.PI* d / c;
    h=1;
    if (Math.abs(a) > 0)     h=Math.sin(a) / g;
    i=e * (4.0 * d / c);
    k=0.54 - 0.46 * Math.cos(l * 2.0 * Math.PI / a);
    j=Math.cos((l - a / 2) * 2.0 * Math.PI* b / c);
    f[l]=j * k * i* h;
  }
  return f;
}

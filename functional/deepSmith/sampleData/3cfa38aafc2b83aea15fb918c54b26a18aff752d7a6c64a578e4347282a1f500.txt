public static int fn_A(int[] a,int b){
  if (a.length < 2)   return 0;
  int c=a.length;
  int[] d=new int[1000000];
  for (int e=0; e < c - 1; e++) {
    for (int f=e + 1; f < c; f++) {
      int g=Math.abs(a[e] - a[f]);
      d[g]=d[g] == 0 ? 1 : d[g] + 1;
    }
  }
  for (int e=0; e < d.length; e++) {
    if (d[e] >= b)     return e;
    b-=d[e];
  }
  return -1;
}

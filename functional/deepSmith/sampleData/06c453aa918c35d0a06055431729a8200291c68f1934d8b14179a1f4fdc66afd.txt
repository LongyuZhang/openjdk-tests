public static Object fn_A(char a){
switch (a) {
case 'B':
    return (byte)0;
case 'C':
  return '\u0000';
case 'D':
return 0.0;
case 'F':
return 0.0f;
case 'I':
return 0;
case 'J':
return 0L;
case 'L':
return null;
case 'S':
return (short)0;
case 'Z':
return false;
case '[':
return null;
default :
return "unknown type represented by " + a;
}
}

public static String fn_A(int a){
switch (a) {
case 0:
    return "Manual Layout";
case 1:
  return "Top-Down (ordered)";
case 2:
return "Top-Down (centered)";
case 3:
return "Top-Down (left-aligned)";
case 4:
return "Left To Right (ordered)";
case 5:
return "Left To Right (curved)";
default :
return "Top-Down (ordered)";
}
}

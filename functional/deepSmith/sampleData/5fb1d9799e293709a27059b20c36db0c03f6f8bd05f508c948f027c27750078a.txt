static void fn_A(int[] a,int b){
  int c=0;
  for (int d=0; d < 31; ++d) {
    c+=a[b + d];
    a[b + d]=c & 255;
    c>>>=8;
  }
  c+=a[b + 31];
  a[b + 31]=c & 127;
  c=19 * (c >>> 7);
  for (int d=0; d < 31; ++d) {
    c+=a[b + d];
    a[b + d]=c & 255;
    c>>>=8;
  }
  c+=a[b + 31];
  a[b + 31]=c;
}

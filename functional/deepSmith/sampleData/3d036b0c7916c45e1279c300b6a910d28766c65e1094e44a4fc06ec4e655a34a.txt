public static final short fn_A(int a,int b,int c){
  int d, e, f;
  d=(((a + 4) > 255 ? 255 : a + 4)) >> 3;
  d=d > 31 ? 31 : d;
  e=(((b + 4) > 255 ? 255 : b + 4)) >> 3;
  e=e > 31 ? 31 : e;
  f=(((c + 4) > 255 ? 255 : c + 4)) >> 3;
  f=f > 31 ? 31 : f;

}

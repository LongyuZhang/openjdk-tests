public static int fn_A(boolean a[],boolean b[],int c){
  if ((a == null) && (b == null)) {
    return -1;
  }
 else   if ((a == null) || (b == null)) {
    return 0;
  }
 else {
    int d=a.length, e=b.length, f=0;
    if (c < 0) {
      c=(d >= e) ? d : e;
    }
    for (f=0; (f < d) && (f < e) && (f < c); f++) {
      if (a[f] != b[f]) {
        return f;
      }
    }
    return (f < c) ? f : -1;
  }
}

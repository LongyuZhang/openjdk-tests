private static boolean fn_A(double a,double b,double c,double d,double e,double f,double g,double h){
  double i, j;
  double k, l;
  i=c - a;
  j=d - b;
  k=g - e;
  l=h - f;
  double m, n;
  m=(-j * (a - e) + i * (b - f)) / (-k * j + i * l);
  n=(k * (b - f) - l * (a - e)) / (-k * j + i * l);
  return m >= 0 && m <= 1 && n >= 0 && n <= 1;
}

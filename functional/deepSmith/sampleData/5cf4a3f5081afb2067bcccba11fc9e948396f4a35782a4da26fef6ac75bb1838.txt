private static Integer fn_A(Object a){
  if (a == null) {
    return null;
  }
  if ((a instanceof Byte)) {
    return Integer.valueOf(((Byte)a).byteValue());
  }
  if ((a instanceof Integer)) {
    return (Integer)a;
  }
  if ((a instanceof Double)) {
    return Integer.valueOf((int)((Double)a).doubleValue());
  }
  if ((a instanceof Float)) {
    return Integer.valueOf((int)((Float)a).floatValue());
  }
  if ((a instanceof Long)) {
    return Integer.valueOf((int)((Long)a).longValue());
  }
  return null;
}

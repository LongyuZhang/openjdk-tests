public static double fn_A(double[] a,double[] b,double c,double d,double e,double f){
  if (a.length != b.length)   throw new IllegalArgumentException("x and y arrays must be same length!");
  int g=a.length;
  double h=0.0;
  for (int i=0; i < a.length; i++) {
    if (Double.isNaN(a[i]) || Double.isNaN(b[i])) {
      g--;
    }
 else {
      h+=(a[i] - c) * (b[i] - d);
    }
  }
  h/=g;
  h/=(e * f);
  return h;
}

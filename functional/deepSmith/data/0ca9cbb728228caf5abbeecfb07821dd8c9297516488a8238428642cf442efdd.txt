public static String fn_A(long a){
  String b;
  if (a < 60) {
    b="\\d{4}-\\d{2}-\\d{2}\\.\\d{2}\\.\\d{2}\\.\\d{2}";
  }
 else   if (a < 3600) {
    b="\\d{4}-\\d{2}-\\d{2}\\.\\d{2}\\.\\d{2}";
  }
 else   if (a < 86400) {
    b="\\d{4}-\\d{2}-\\d{2}\\.\\d{2}";
  }
 else {
    b="\\d{4}-\\d{2}-\\d{2}";
  }
  return b;
}

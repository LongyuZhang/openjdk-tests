public static void main(String[] a){
  int[][] b={{1,2,3},{4,5,6}};
  int[][] c=new int[b[0].length][b.length];
  for (int d=0; d < b[0].length; d++) {
    for (int e=0; e < b.length; e++) {
      c[d][e]=b[e][d];
      System.out.print(c[d][e] + " ");
      if (e == b.length - 1)       System.out.print("\n");
    }
  }
}

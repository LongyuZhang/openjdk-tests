public static void fn_A(byte[] a,int b,long c) throws IllegalArgumentException {
  if (a.length - b < 8) {
    throw new IllegalArgumentException("not enough space to store long");
  }
  a[b]=(byte)(c >> 56);
  a[b + 1]=(byte)(c >> 48);
  a[b + 2]=(byte)(c >> 40);
  a[b + 3]=(byte)(c >> 32);
  a[b + 4]=(byte)(c >> 24);
  a[b + 5]=(byte)(c >> 16);
  a[b + 6]=(byte)(c >> 8);
  a[b + 7]=(byte)c;
}

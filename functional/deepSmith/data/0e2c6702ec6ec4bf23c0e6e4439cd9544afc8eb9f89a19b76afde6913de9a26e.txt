private static int fn_A(char a){
  if (a >= 'A' && a <= 'Z')   return ((int)a) - 65;
 else   if (a >= 'a' && a <= 'z')   return ((int)a) - 97 + 26;
 else   if (a >= '0' && a <= '9')   return ((int)a) - 48 + 26 + 26;
 else {
switch (a) {
case '+':
      return 62;
case '/':
    return 63;
case '=':
  return 0;
default :
throw new RuntimeException("unexpected code: " + a);
}
}
}

public static int fn_A(int a,int b){
  int[][] c=new int[a][b];
  for (int d=0; d < a; d++) {
    for (int e=0; e < b; e++) {
      if (d == 0 && e == 0)       c[d][e]=1;
 else       if (d == 0)       c[d][e]=c[d][e - 1];
 else       if (e == 0)       c[d][e]=c[d - 1][e];
 else       c[d][e]=c[d - 1][e] + c[d][e - 1];
    }
  }
  return c[a - 1][b - 1];
}

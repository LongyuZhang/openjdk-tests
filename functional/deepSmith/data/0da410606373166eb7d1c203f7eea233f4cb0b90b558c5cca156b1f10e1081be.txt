public static byte[] fn_A(final long a){
  final byte[] b=new byte[8];
  b[7]=(byte)(a & 0xff);
  b[6]=(byte)(a >> 8 & 0xff);
  b[5]=(byte)(a >> 16 & 0xff);
  b[4]=(byte)(a >> 24 & 0xff);
  b[3]=(byte)(a >> 32 & 0xff);
  b[2]=(byte)(a >> 40 & 0xff);
  b[1]=(byte)(a >> 48 & 0xff);
  b[0]=(byte)(a >> 56 & 0xff);
  return b;
}

public static void fn_A(int a[],boolean b){
  for (int c=0; c < a.length - 1; ++c) {
    int d=c;
    for (int e=c + 1; e < a.length; ++e) {
      if ((b && a[e] < a[d]) || (!b && a[e] > a[d])) {
        d=e;
      }
    }
    if (d != c) {
      int f=a[c];
      a[c]=a[d];
      a[d]=f;
    }
  }
}

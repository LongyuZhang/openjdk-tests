public static final byte fn_A(int a){
  if (a >= 15) {
    return 7;
  }
  if (a == 14) {
    return 7;
  }
  if (a == 13) {
    return 6;
  }
  if (a >= 13) {
    return 6;
  }
  if (a >= 11 && a <= 12) {
    return 5;
  }
  if (a >= 9 && a <= 10) {
    return 4;
  }
  if (a >= 7 && a <= 8) {
    return 3;
  }
  if (a >= 5 && a <= 6) {
    return 2;
  }
  if (a >= 3 && a <= 4) {
    return 1;
  }
  return 0;
}

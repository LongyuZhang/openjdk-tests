public static int fn_A(char[] a,char[] b,int c){
  if (a.length == 0 && b.length - c > 0)   return -1;
  for (int d=0; d < a.length; d++) {
    if (c + d >= b.length)     return d + 1;
    if (a[d] != b[c + d]) {
      if (a[d] < b[c + d])       return -(d + 1);
      return d + 1;
    }
  }
  return 0;
}

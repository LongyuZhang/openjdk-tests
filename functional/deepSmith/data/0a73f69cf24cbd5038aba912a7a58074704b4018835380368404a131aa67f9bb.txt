public static double[][] fn_A(float a){
  double[][] b=new double[5][3];
  float c=a / 2.0f;
  b[0][0]=0;
  b[0][1]=0;
  b[0][2]=-c * 1.2;
  b[1][0]=1;
  b[1][1]=c;
  b[1][2]=0;
  b[2][0]=1;
  b[2][1]=0;
  b[2][2]=c * 1.2;
  b[3][0]=1;
  b[3][1]=-c;
  b[3][2]=0;
  b[4][0]=1;
  b[4][1]=0;
  b[4][2]=-c * 1.2;
  return b;
}

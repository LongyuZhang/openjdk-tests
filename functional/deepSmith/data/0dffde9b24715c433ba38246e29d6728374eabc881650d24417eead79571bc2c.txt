private static int fn_A(final int a,final int b,final int c){
  assert(a >= 2);
  assert(a <= b);
  assert(a <= c);
  int d=0;
  int e=1;
  for (int f=1; f <= (b + c - 2) / 2; ++f) {
    if (e < b)     ++e;
    if (e >= a) {
      final int g=(2 * f == b + c - 2) ? 1 : 2;
      d+=g * 2 * (e - a + 1);
    }
  }
  return d;
}

private static boolean fn_A(double[][] a,double[][] b){
  if (a == null) {
    return (b == null);
  }
  if (b == null) {
    return false;
  }
  if (a.length != b.length) {
    return false;
  }
  for (int c=0; c < a.length; c++) {
    if (!Arrays.equals(a[c],b[c])) {
      return false;
    }
  }
  return true;
}

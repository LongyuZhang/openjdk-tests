public static int[][] fn_A(int[][] a,int[][] b){
  int c=a.length;
  int[][] d=new int[c][c];
  for (int e=0; e < c; e++) {
    for (int f=0; f < c; f++) {
      int g=0;
      for (int h=0; h < c; h++) {
        final int i=a[e][h];
        final int j=b[h][e];
        g+=i * j;
      }
      d[e][f]=g;
    }
  }
  return d;
}

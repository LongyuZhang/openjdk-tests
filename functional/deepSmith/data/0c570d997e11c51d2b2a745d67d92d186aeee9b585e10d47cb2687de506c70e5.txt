public static int[] fn_A(int[] a){
  int b=(a[a.length - 1] + 1 >= 10) ? 1 : 0;
  a[a.length - 1]=(a[a.length - 1] + 1) % 10;
  for (int c=a.length - 2; c >= 0; --c) {
    int d=a[c];
    a[c]=(d + b) % 10;
    b=(d + b >= 10) ? 1 : 0;
  }
  if (b == 1) {
    int[] e=new int[a.length + 1];
    e[0]=1;
    for (int c=0; c < a.length; ++c) {
      e[c + 1]=a[c];
    }
    return e;
  }
  return a;
}

public static Integer fn_A(Object a){
  if (a instanceof Double)   return ((Double)a).intValue();
 else   if (a instanceof Float)   return ((Float)a).intValue();
 else   if (a instanceof BigDecimal)   return ((BigDecimal)a).toBigInteger().intValue();
 else   if (a instanceof Long)   return ((Long)a).intValue();
 else   if (a instanceof Integer)   return ((Integer)a);
 else   if (a instanceof Short)   return ((Short)a).intValue();
 else   if (a instanceof Byte)   return ((Byte)a).intValue();
 else   if (a instanceof Boolean)   return ((Boolean)a) ? 1 : 0;
 else {
    return null;
  }
}

public static double[] fn_A(double[] a,double[][] b){
  int c=b.length;
  int d=b[0].length;
  if (a.length != c) {
    throw new RuntimeException("Illegal matrix dimensions.");
  }
  double[] e=new double[d];
  for (int f=0; f < d; f++) {
    for (int g=0; g < c; g++) {
      e[f]+=(b[g][f] * a[g]);
    }
  }
  return e;
}

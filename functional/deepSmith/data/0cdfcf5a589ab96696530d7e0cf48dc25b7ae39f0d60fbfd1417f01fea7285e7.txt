public static Double fn_A(Object a){
  if (a instanceof Double)   return (Double)a;
 else   if (a instanceof Float)   return ((Float)a).doubleValue();
 else   if (a instanceof BigDecimal)   return ((BigDecimal)a).doubleValue();
 else   if (a instanceof Long)   return ((Long)a).doubleValue();
 else   if (a instanceof Integer)   return ((Integer)a).doubleValue();
 else   if (a instanceof Short)   return ((Short)a).doubleValue();
 else   if (a instanceof Byte)   return ((Byte)a).doubleValue();
 else {
    return null;
  }
}

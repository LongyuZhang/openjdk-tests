public static final int fn_A(long a){
  int b=0;
  if ((256 & a) != 0) {
    b=0 | 256;
  }
 else   if ((1 & a) != 0) {
    b=0 | 1;
  }
  if ((512 & a) != 0) {
    b|=512;
  }
 else   if ((2 & a) != 0) {
    b|=2;
  }
  if ((1024 & a) != 0) {
    return b | 1024;
  }
  if ((4 & a) != 0) {
    return b | 4;
  }
  return b;
}

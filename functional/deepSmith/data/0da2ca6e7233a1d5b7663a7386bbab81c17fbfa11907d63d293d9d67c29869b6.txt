public static void fn_A(boolean a[][],int b[],int c){
  int d=b.length;
  int e=c;
  System.out.print("The subset is ");
  while (true) {
    if (a[d - 1][e]) {
      d=d - 1;
      continue;
    }
 else {
      System.out.print(b[d - 1] + " ");
      e=e - b[d - 1];
      if (e <= 0)       break;
    }
  }
  System.out.println();
}

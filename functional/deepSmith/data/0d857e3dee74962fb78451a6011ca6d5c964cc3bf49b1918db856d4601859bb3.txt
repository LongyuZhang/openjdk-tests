public static int fn_A(long a){
  int b=0;
  if ((a & 0xffffffff00000000l) != 0) {
    a>>>=64;
    b=64;
  }
  if (a >= 4294967296l) {
    a>>>=32;
    b=32;
  }
  if (a >= 65536) {
    a>>>=16;
    b=16;
  }
  if (a >= 256) {
    a>>>=8;
    b+=8;
  }
  if (a >= 16) {
    a>>>=4;
    b+=4;
  }
  if (a >= 4) {
    a>>>=2;
    b+=2;
  }
  if (a >= 2) {
    b++;
  }
  return b;
}
